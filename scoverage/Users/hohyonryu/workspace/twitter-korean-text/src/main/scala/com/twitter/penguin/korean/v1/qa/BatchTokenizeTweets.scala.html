27 /**
28  * Batch tokenize a file of Korean tweets.
29  *
30  * usage: ./pants goal run src/scala/com/twitter/penguin/korean/qa:batch_tokenize_tweets
31  * --jvm-run-args="~/korean_201407_100"
32  */
33 object BatchTokenizeTweets {
34 
35   case class ParseTime(time: Long, chunk: String)
36 
37   val LOG = Logger.getLogger(getClass.getSimpleName)
38   val VERBOSE = true
39 
40   def main(args: Array[String]) {
41     if (args.length != 1) {
42       println("The first arg should be an input file of Korean tweets.")
43       return
44     }
45     val parseTimesAll = Source.fromFile(args(0)).getLines().foldLeft(List[ParseTime]()) {
46       case (l: List[ParseTime], line: String) if line.trim.length > 5 =>
47         if (VERBOSE) println(line.trim)
48 
49         val t0 = System.currentTimeMillis()
50         val parsed = TwitterKoreanProcessor.tokenize(line)
51         val t1 = System.currentTimeMillis()
52 
53         if (VERBOSE) {
54           println(parsed.map {
55             case t if t.unknown => t.text.toString + t.pos + "*"
56             case t => t.text + t.pos.toString
57           }.mkString(" "))
58 
59           println()
60         }
61         ParseTime(t1 - t0, line.trim) :: l
62       case (l: List[ParseTime], line: String) => l
63     }
64 
65     val loadingTime = parseTimesAll.last
66 
67     LOG.log(Level.INFO, "The first one \"%s\" took %d ms including the loading time.".format(loadingTime.chunk, loadingTime.time))
68 
69     val parseTimes = parseTimesAll.init
70 
71     val averageTweetLength = parseTimes.map(_.chunk.length).sum.toDouble / parseTimes.size
72 
73     val averageTime = parseTimes.map(_.time).sum.toDouble / parseTimes.size
74     val maxItem = parseTimes.maxBy(_.time)
75 
76     LOG.log(Level.INFO, ("Parsed %d items. \n" +
77         "       Total time: %d s \n" +
78         "       Average tweet length: %.2f chars \n" +
79         "       Average time per tweet: %.2f ms \n" +
80         "       Max time: %d ms, %s\n" +
81         "       Parsed: %s"
82         ).format(
83           parseTimes.size,
84           parseTimes.map(_.time).sum / 1000,
85           averageTweetLength,
86           averageTime,
87           maxItem.time,
88           maxItem.chunk,
89           TwitterKoreanProcessor.tokenize(maxItem.chunk).map {
90             case t if t.unknown => t.text.toString + t.pos + "*"
91             case t => t.text + t.pos.toString
92           }.mkString(" ")
93         ))
94   }
95 }
