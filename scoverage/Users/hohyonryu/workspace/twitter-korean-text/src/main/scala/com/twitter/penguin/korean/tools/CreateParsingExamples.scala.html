37     val parsedPairs = readFileByLineFromResources("example_chunks.txt").flatMap {

39         val chunk = line.trim
40         val parsed = tokenize(chunk)
41         Some(ParsingExample(chunk, parsed))
42       case line => None
43     }.toSet
44 
45 
46     val outputFile: String = "src/test/resources/com/twitter/penguin/korean/util/current_parsing.txt"

50     val out = new FileOutputStream(outputFile)
51     parsedPairs.foreach {
52       p =>
53         out.write(p.text.getBytes)
54         out.write("\t".getBytes)
55         out.write(p.parse.mkString("/").getBytes)
56         out.write("\n".getBytes)
57     }
