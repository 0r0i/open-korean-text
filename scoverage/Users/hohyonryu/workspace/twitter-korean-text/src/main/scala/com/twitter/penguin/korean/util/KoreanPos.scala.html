21 /**
22  * Korean Part-of-Speech
23  *
24  * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
25  * V Verb: 동사 (하, 먹, 자, 차)
26  * J Adjective: 형용사 (예쁘다, 크다, 작다)
27  * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
28  * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
29  * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
30  *
31  * C Conjunction: 접속사
32  *
33  * j SubstantiveJosa: 조사 (의, 에, 에서)
34  * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
35  * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
36  * r PreEomi: 선어말어미 (었)
37  *
38  * p NounPrefix: 접두사 ('초'대박)
39  * v VerbPrefix: 동사 접두어 ('쳐'먹어)
40  * s Suffix: 접미사 (~적)
41  *
42  * f Foreign: 한글이 아닌 문자들
43  *
44  * 지시사는 Derterminant로 대체하기로 함
45  * Derterminant is used for demonstratives.
46  *
47  * Korean: Korean chunk (candidate for parsing)
48  * Foreign: Mixture of non-Korean strings
49  * Number: 숫자
50  * Emotion: Korean Single Character Emotions (ㅋㅋㅋㅋ, ㅎㅎㅎㅎ, ㅠㅜㅠㅜ)
51  * Alpha: Alphabets 알파벳
52  * Punctuation: 문장부호
53  * Hashtag: Twitter Hashtag 해쉬태그 #Korean
54  * ScreenName: Twitter username (@nlpenguin)
55  *
56  * Unkown: Could not parse the string.
57  */
58 object KoreanPos extends Enumeration {
59   type KoreanPos = Value
60 
61   // Word leved POS
62   val Noun, Verb, Adjective,
63   Adverb, Determiner, Exclamation,
64   Josa, Eomi, PreEomi, Conjunction,
65   NounPrefix, VerbPrefix, Suffix, Unknown,
66 
67   // Chunk level POS
68   Korean, Foreign, Number, KoreanParticle, Alpha,
69   Punctuation, Hashtag, ScreenName,
70   Email, URL, CashTag,
71 
72   // Functional POS
73   Space, Others = Value
74 
75   val OtherPoses = Set(Korean, Foreign, Number, KoreanParticle, Alpha,
76     Punctuation, Hashtag, ScreenName,
77     Email, URL, CashTag)
78 
79   val shortCut = Map(
80     'N' -> Noun,
81     'V' -> Verb,
82     'J' -> Adjective,
83     'A' -> Adverb,
84     'D' -> Determiner,
85     'E' -> Exclamation,
86     'C' -> Conjunction,
87 
88     'j' -> Josa,
89     'e' -> Eomi,
90     'r' -> PreEomi,
91     'p' -> NounPrefix,
92     'v' -> VerbPrefix,
93     's' -> Suffix,
94 
95     'a' -> Alpha,
96     'n' -> Number,
97 
98     'o' -> Others
99   )
100 
101   case class KoreanPosTrie(curPos: KoreanPos, nextTrie: List[KoreanPosTrie], ending: Option[KoreanPos])
102 
103   val selfNode = KoreanPosTrie(null, null, ending = None)
104 
105   protected[korean] def buildTrie(s: String, ending_pos: KoreanPos): List[KoreanPosTrie] = {
106     def isFinal(rest: String): Boolean = {
107       val isNextOptional = rest.foldLeft(true) {
108         case (output: Boolean, c: Char) if c == '+' || c == '1' => false
109         case (output: Boolean, c: Char) => output
110       }
111       rest.length == 0 || isNextOptional
112     }
113 
114     if (s.length < 2) {
115       return List()
116     }
117 
118     val pos = shortCut(s.charAt(0))
119     val rule = s.charAt(1)
120     val rest = if (s.length > 1) {
121       s.slice(2, s.length)
122     } else {
123       ""
124     }
125 
126     val end: Option[KoreanPos] = if (isFinal(rest)) Some(ending_pos) else None
127 
128     rule match {
129       case '+' =>
130         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest, ending_pos), end))
131       case '*' =>
132         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest, ending_pos), end)) ++ buildTrie(rest, ending_pos)
133       case '1' =>
134         List(KoreanPosTrie(pos, buildTrie(rest, ending_pos), end))
135       case '0' =>
136         List(KoreanPosTrie(pos, buildTrie(rest, ending_pos), end)) ++ buildTrie(rest, ending_pos)
137     }
138   }
139 
140   protected[korean] def getTrie(sequences: Map[String, KoreanPos]): List[KoreanPosTrie] =
141     sequences.foldLeft(List[KoreanPosTrie]()) {
142       case (results: List[KoreanPosTrie], (s: String, ending_pos: KoreanPos)) =>
143         buildTrie(s, ending_pos) ::: results
144     }
145 
146   val Predicates = Set(Verb, Adjective)
147 }
