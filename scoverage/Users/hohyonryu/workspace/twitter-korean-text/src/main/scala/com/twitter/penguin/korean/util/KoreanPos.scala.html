21 /**
22  * Korean Part-of-Speech
23  *
24  * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
25  * V Verb: 동사 (하, 먹, 자, 차)
26  * J Adjective: 형용사 (예쁘다, 크다, 작다)
27  * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
28  * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
29  * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
30  *
31  * C Conjunction: 접속사
32  *
33  * j SubstantiveJosa: 조사 (의, 에, 에서)
34  * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
35  * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
36  * r PreEomi: 선어말어미 (었)
37  *
38  * p NounPrefix: 접두사 ('초'대박)
39  * v VerbPrefix: 동사 접두어 ('쳐'먹어)
40  * s Suffix: 접미사 (~적)
41  *
42  * f Foreign: 한글이 아닌 문자들
43  *
44  * 지시사는 Derterminant로 대체하기로 함
45  * Derterminant is used for demonstratives.
46  *
47  * Korean: Korean chunk (candidate for parsing)
48  * Foreign: Mixture of non-Korean strings
49  * Number: 숫자
50  * Emotion: Korean Single Character Emotions (ㅋㅋㅋㅋ, ㅎㅎㅎㅎ, ㅠㅜㅠㅜ)
51  * Alpha: Alphabets 알파벳
52  * Punctuation: 문장부호
53  * Hashtag: Twitter Hashtag 해쉬태그 #Korean
54  * ScreenName: Twitter username (@nlpenguin)
55  *
56  * Unkown: Could not parse the string.
57  */
58 object KoreanPos extends Enumeration {
59   type KoreanPos = Value
60 
61   // Word leved POS
62   val Noun, Verb, Adjective,
63   Adverb, Determiner, Exclamation,
64   Josa, Eomi, PreEomi, Conjunction,
65   NounPrefix, VerbPrefix, Suffix, Unknown,
66 
67   // Chunk level POS
68   Korean, Foreign, Number, KoreanParticle, Alpha,
69   Punctuation, Hashtag, ScreenName,
70   Email, URL, CashTag,
71 
72   // Functional POS
73   Space, Others,
74 
75   ProperNoun = Value
76 
77   val OtherPoses = Set(Korean, Foreign, Number, KoreanParticle, Alpha,
78     Punctuation, Hashtag, ScreenName,
79     Email, URL, CashTag)
80 
81   val shortCut = Map(
82     'N' -> Noun,
83     'V' -> Verb,
84     'J' -> Adjective,
85     'A' -> Adverb,
86     'D' -> Determiner,
87     'E' -> Exclamation,
88     'C' -> Conjunction,
89 
90     'j' -> Josa,
91     'e' -> Eomi,
92     'r' -> PreEomi,
93     'p' -> NounPrefix,
94     'v' -> VerbPrefix,
95     's' -> Suffix,
96 
97     'a' -> Alpha,
98     'n' -> Number,
99 
100     'o' -> Others
101   )
102 
103   case class KoreanPosTrie(curPos: KoreanPos, nextTrie: List[KoreanPosTrie], ending: Option[KoreanPos])
104 
105   val selfNode = KoreanPosTrie(null, null, ending = None)
106 
107   protected[korean] def buildTrie(s: String, ending_pos: KoreanPos): List[KoreanPosTrie] = {
108     def isFinal(rest: String): Boolean = {
109       val isNextOptional = rest.foldLeft(true) {
110         case (output: Boolean, c: Char) if c == '+' || c == '1' => false
111         case (output: Boolean, c: Char) => output
112       }
113       rest.length == 0 || isNextOptional
114     }
115 
116     if (s.length < 2) {
117       return List()
118     }
119 
120     val pos = shortCut(s.charAt(0))
121     val rule = s.charAt(1)
122     val rest = if (s.length > 1) {
123       s.slice(2, s.length)
124     } else {
125       ""
126     }
127 
128     val end: Option[KoreanPos] = if (isFinal(rest)) Some(ending_pos) else None
129 
130     rule match {
131       case '+' =>
132         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest, ending_pos), end))
133       case '*' =>
134         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest, ending_pos), end)) ++ buildTrie(rest, ending_pos)
135       case '1' =>
136         List(KoreanPosTrie(pos, buildTrie(rest, ending_pos), end))
137       case '0' =>
138         List(KoreanPosTrie(pos, buildTrie(rest, ending_pos), end)) ++ buildTrie(rest, ending_pos)
139     }
140   }
141 
142   protected[korean] def getTrie(sequences: Map[String, KoreanPos]): List[KoreanPosTrie] =
143     sequences.foldLeft(List[KoreanPosTrie]()) {
144       case (results: List[KoreanPosTrie], (s: String, ending_pos: KoreanPos)) =>
145         buildTrie(s, ending_pos) ::: results
146     }
147 
148   val Predicates = Set(Verb, Adjective)
149 }
