31 object CreatePhraseExtractionExamples extends Runnable {
32 
33   case class PhraseExample(text: String, phrases: Seq[KoreanPhrase])
34 
35   def run {
36     System.err.println("Reading the sample tweets..")
37 
38     val phrasePairs = readFileByLineFromResources("example_tweets.txt").flatMap {
39       case line if line.length > 0 =>
40         val chunk = line.trim
41         val normalized = KoreanNormalizer.normalize(chunk)
42         val tokens = tokenize(normalized)
43         val phrases = extractPhrases(tokens)
44         Some(PhraseExample(chunk, phrases))
45       case line => None
46     }.toSet
47 
48 
49     val outputFile: String = "src/test/resources/com/twitter/penguin/korean/util/current_phrases.txt"
50 
51     System.err.println("Writing the new phrases to " + outputFile)
52 
53     val out = new FileOutputStream(outputFile)
54     phrasePairs.foreach {
55       p =>
56         out.write(p.text.getBytes)
57         out.write("\t".getBytes)
58         out.write(p.phrases.mkString("/").getBytes)
59         out.write("\n".getBytes)
60     }
61     out.close()
62 
63     System.err.println("Testing the new phrases " + outputFile)
64   }
65 }
