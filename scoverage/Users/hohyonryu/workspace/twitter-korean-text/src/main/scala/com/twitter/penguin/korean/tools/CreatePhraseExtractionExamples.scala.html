36     val phrasePairs = readFileByLineFromResources("example_tweets.txt").flatMap {

39         val phrases = extractPhrases(chunk)

42     }.toSet

45     val outputFile: String = "src/test/resources/com/twitter/penguin/korean/util/current_phrases.txt"

52         out.write(p.text.getBytes)

54         out.write(p.phrases.mkString("/").getBytes)
55         out.write("\n".getBytes)
