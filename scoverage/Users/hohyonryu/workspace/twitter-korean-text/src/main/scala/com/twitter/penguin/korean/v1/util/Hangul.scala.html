72   /**
73    * Decompose a Korean character to onset(초성), vowel(중성), and coda(종성).
74    *
75    * @param c A Korean character
76    * @return (onset: Char, vowel: Char, coda: Char)
77    */
78   def decomposeHangul(c: Char): (Char, Char, Char) = {
79     require(!ONSET_MAP.contains(c) && !VOWEL_MAP.contains(c) && !CODA_MAP.contains(c),
80       "Input character is not a valid Korean character")
81 
82     val u = c - HANGUL_BASE
83     (ONSET_LIST(u / ONSET_BASE),
84         VOWEL_LIST((u % ONSET_BASE) / VOWEL_BASE),
85         CODA_LIST(u % VOWEL_BASE))
86   }
87 
88   /**
89    * Check if a Korean character has a coda.
90    *
91    * @param c A Korean character
92    * @return true if the character has a coda.
93    */
94   def hasCoda(c: Char): Boolean = (c - HANGUL_BASE) % VOWEL_BASE > 0
95 
96   /**
97    * Compose a Korean character from the provided onset(초성), vowel(중성), and coda(종성).
98    *
99    * @param onset 초성
100    * @param vowel 중성
101    * @param coda 종성
102    * @return A Korean character
103    */
104   def composeHangul(onset: Char, vowel: Char, coda: Char = ' '): Char = {
105     require(onset != ' ' && vowel != ' ', "Input characters are not valid")
106 
107     (HANGUL_BASE +
108         (ONSET_MAP(onset) * ONSET_BASE) +
109         (VOWEL_MAP(vowel) * VOWEL_BASE) +
110         CODA_MAP(coda)).toChar
111   }
