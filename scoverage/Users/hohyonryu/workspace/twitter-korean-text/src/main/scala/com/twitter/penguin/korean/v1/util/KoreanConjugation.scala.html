31   // ㅋ, ㅎ for 잨ㅋㅋㅋㅋ 잔댛ㅎㅎㅎㅎ
32   private[this] val CODAS_COMMON = Seq('ㅂ', 'ㅆ', 'ㄹ', 'ㄴ', 'ㅁ')
33   private[this] val CODAS_NO_PAST = Seq('ㅂ', 'ㄹ', 'ㄴ', 'ㅁ')

35   private[this] val CODAS_SLANG_CONSONANT = Seq('ㅋ', 'ㅎ')
36   private[this] val CODAS_SLANG_VOWEL = Seq('ㅜ', 'ㅠ')

38   private[this] val PRE_EOMI_COMMON = "노러도게면네더던든고거구기긴길다자재죠져지진질겠".toSeq

41   private[this] val PRE_EOMI_1_3 = "아았".toSeq

48   private[this] val PRE_EOMI_5 = "은".toSeq

57   private[this] def addPreEomi(lastChar: Char, charsToAdd: Seq[Char]): Seq[String] = {
58     charsToAdd.map(lastChar + _.toString)

61   /**
62    * Conjugate adjectives and verbs.
63    *
64    * @param words Set of adjectives or verbs.
65    * @param isAdjective True if the set contains adjectives.
66    * @return CharArraySet with conjugated words.
67    */
68   protected[korean] def conjugatePredicates(words: Set[String], isAdjective: Boolean = false): CharArraySet = {
69     lazy val expanded = words.flatMap { word: String =>
70       val init = word.init
71       val lastChar = word.last
72       val lastCharString = lastChar.toString
73       val lastCharDecomposed = decomposeHangul(lastChar)
74 
75       val expandedLast: Seq[String] = lastCharDecomposed match {
76         // 하다, special case
77         case ('ㅎ', 'ㅏ', ' ') =>
78           val endings = if (isAdjective) Seq("합", "해", "히", "") else Seq("합", "해")
79           addPreEomi(lastChar, PRE_EOMI_COMMON ++ PRE_EOMI_2 ++ PRE_EOMI_6 ++ PRE_EOMI_RESPECT) ++
80               CODAS_COMMON.map {
81                 case c: Char if c == 'ㅆ' => composeHangul('ㅎ', 'ㅐ', c).toString
82                 case c: Char => composeHangul('ㅎ', 'ㅏ', c).toString
83               } ++ addPreEomi('하', PRE_EOMI_VOWEL ++ PRE_EOMI_1_5 ++ PRE_EOMI_6) ++
84               addPreEomi('해', PRE_EOMI_1_1) ++ endings
85 
86         // 쏘다
87         case (o: Char, 'ㅗ', ' ') =>
88           addPreEomi(lastChar, PRE_EOMI_VOWEL ++ PRE_EOMI_2 ++ PRE_EOMI_1_3 ++ PRE_EOMI_6) ++
89               CODAS_NO_PAST.map(composeHangul(o, 'ㅗ', _).toString) ++
90               Seq(composeHangul(o, 'ㅘ', ' ').toString,
91                 composeHangul(o, 'ㅘ', 'ㅆ').toString,
92                 lastCharString)
93 
94         // 맞추다, 겨누다, 재우다,
95         case (o: Char, 'ㅜ', ' ') =>
96           addPreEomi(lastChar, PRE_EOMI_VOWEL ++ PRE_EOMI_1_2 ++ PRE_EOMI_6) ++
97               CODAS_NO_PAST.map(composeHangul(o, 'ㅜ', _).toString) ++
98               Seq(composeHangul(o, 'ㅝ').toString,
99                 composeHangul(o, 'ㅝ', 'ㅆ').toString,
100                 lastCharString)
101 
102 
103         // 치르다, 구르다, 굴르다, 뜨다, 모으다, 고르다, 골르다
104         case (o: Char, 'ㅡ', ' ') =>
105           CODAS_NO_PAST.map(composeHangul(o, 'ㅡ', _).toString) ++
106               Seq(composeHangul(o, 'ㅝ').toString,
107                 composeHangul(o, 'ㅓ').toString,
108                 composeHangul(o, 'ㅏ').toString,
109                 composeHangul(o, 'ㅝ', 'ㅆ').toString,
110                 composeHangul(o, 'ㅓ', 'ㅆ').toString,
111                 composeHangul(o, 'ㅏ', 'ㅆ').toString,
112                 lastCharString)
113 
114         // 사귀다
115         case (o: Char, 'ㅟ', ' ') =>
116           CODAS_NO_PAST.map(composeHangul(o, 'ㅟ', _).toString) ++
117               Seq(composeHangul(o, 'ㅕ', ' ').toString, composeHangul(o, 'ㅕ', 'ㅆ').toString) ++
118               Seq(lastCharString)
119 
120         // 마시다, 엎드리다, 치다
121         case (o: Char, 'ㅣ', ' ') =>
122           CODAS_NO_PAST.map(composeHangul(o, 'ㅣ', _).toString) ++
123               Seq(composeHangul(o, 'ㅕ', ' ').toString,
124                 composeHangul(o, 'ㅕ', 'ㅆ').toString,
125                 lastCharString)
126 
127         // 꿰다, 꾀다
128         case (o: Char, v: Char, ' ') if v == 'ㅞ' || v == 'ㅚ' || v == 'ㅙ' =>
129           CODAS_COMMON.map(composeHangul(o, v, _).toString) ++
130               Seq(lastCharString)
131 
132 
133         // All other vowel endings: 둘러서다, 켜다, 세다, 캐다
134         case (o: Char, v: Char, ' ') =>
135           CODAS_COMMON.map(composeHangul(o, v, _).toString) ++
136               addPreEomi(lastChar, PRE_EOMI_VOWEL ++ PRE_EOMI_1_1 ++ PRE_EOMI_6) ++
137               Seq(lastCharString)
138 
139         // 만들다, 알다, 풀다
140         case (o: Char, v: Char, 'ㄹ') if v == 'ㅡ' || v == 'ㅏ' || v == 'ㅜ' =>
141           addPreEomi(composeHangul(o, v, ' '), PRE_EOMI_2 ++ PRE_EOMI_3 ++ PRE_EOMI_RESPECT) ++
142               Seq(composeHangul(o, v, 'ㄻ').toString,
143                 composeHangul(o, v, 'ㄴ').toString,
144                 composeHangul(o, v, ' ').toString,
145                 lastCharString)
146 
147         // 낫다
148         case (o: Char, 'ㅏ', 'ㅅ') =>
149           addPreEomi(composeHangul(o, 'ㅏ'), PRE_EOMI_3 ++ PRE_EOMI_4 ++ PRE_EOMI_5) ++
150               Seq(lastCharString)
151 
152         // 묻다
153         case ('ㅁ', 'ㅜ', 'ㄷ') =>
154           Seq(composeHangul('ㅁ', 'ㅜ', 'ㄹ').toString,
155             lastCharString)
156 
157         // 붇다
158         case (o: Char, 'ㅜ', 'ㄷ') =>
159           addPreEomi(composeHangul(o, 'ㅜ', ' '), PRE_EOMI_1_2 ++ PRE_EOMI_1_4 ++ PRE_EOMI_4 ++ PRE_EOMI_5) ++
160               Seq(composeHangul(o, 'ㅜ', 'ㄹ').toString,
161                 lastCharString)
162 
163         // 눕다
164         case (o: Char, 'ㅜ', 'ㅂ') =>
165           addPreEomi(composeHangul(o, 'ㅜ', ' '), PRE_EOMI_1_4 ++ PRE_EOMI_4 ++ PRE_EOMI_5) ++
166               Seq(lastCharString)
167 
168         // 간지럽다, 갑작스럽다 -> 갑작스런
169         case (o: Char, 'ㅓ', 'ㅂ') if isAdjective =>
170           addPreEomi(composeHangul(o, 'ㅓ', ' '), PRE_EOMI_1_4 ++ PRE_EOMI_7) ++
171               Seq(composeHangul(o, 'ㅓ', ' ').toString, composeHangul(o, 'ㅓ', 'ㄴ').toString, lastCharString)
172 
173         // 아름답다, 가볍다, 덥다, 간지럽다
174         case (o: Char, v: Char, 'ㅂ') if isAdjective =>
175           addPreEomi(composeHangul(o, v, ' '), PRE_EOMI_1_4 ++ PRE_EOMI_7) ++
176               Seq(composeHangul(o, v, ' ').toString, lastCharString)
177 
178         // 놓다
179         case (o: Char, 'ㅗ', 'ㅎ') =>
180           CODAS_COMMON.map(composeHangul(o, 'ㅗ', _).toString) ++
181               Seq(composeHangul(o, 'ㅘ', ' ').toString, composeHangul(o, 'ㅗ', ' ').toString, lastCharString)
182 
183         // 파랗다, 퍼렇다
184         case (o: Char, v: Char, 'ㅎ') if isAdjective =>
185           CODAS_COMMON.map(composeHangul(o, v, _).toString) ++
186               Seq(composeHangul(o, v, ' ').toString, lastCharString)
187 
188         // 1 char with coda, 작다
189         case (o: Char, v: Char, c: Char) if word.length == 1 =>
190           addPreEomi(lastChar, PRE_EOMI_COMMON ++ PRE_EOMI_1_2 ++ PRE_EOMI_1_3 ++ PRE_EOMI_4 ++ PRE_EOMI_5) ++
191               Seq(lastCharString)
192 
193         // 부여잡다, 얻어맞다, 얻어먹다
194         case _ =>
195           Seq(lastCharString)
196 
197       }
198 
199       expandedLast.map(init + _)
200     }
201 
202     lazy val newSet = newCharArraySet
203     newSet.addAll(expanded)
204     newSet
205   }

208   /**
209    * Adds intentional typos at verb endings.
210    * 가래 -> 가랰ㅋㅋㅋㅋㅋ
211    * 먹 -> 머구ㅜㅜㅜㅜㅜ
212    * 닭 -> 달규ㅠㅠㅠ
213    *
214    * @param word Input string.
215    * @return Expanded words.
216    */
217   def addTypoEndings(word: String, applyToOneChar: Boolean = true): Seq[String] = {
218     decomposeHangul(word.last) match {
219       case (o: Char, v: Char, ' ') if applyToOneChar || word.length > 1 =>
220         Seq(word) ++
221             CODAS_SLANG_CONSONANT.map(word.init + composeHangul(o, v, _))
222       case (o: Char, v: Char, c: Char) if DOUBLE_CODAS.contains(c) && word.length > 1 =>
223         val doubleCoda = DOUBLE_CODAS(c)
224         Seq(word) ++
225             CODAS_SLANG_VOWEL.map(word.init + composeHangul(o, v, doubleCoda.first) + composeHangul(doubleCoda.second, _))
226       case (o: Char, v: Char, c: Char) if word.length > 1 =>
227         Seq(word) ++
228             CODAS_SLANG_VOWEL.map(word.init + composeHangul(o, v, ' ') + composeHangul(c, _))
229       case _ => Seq(word)
230     }
231   }
