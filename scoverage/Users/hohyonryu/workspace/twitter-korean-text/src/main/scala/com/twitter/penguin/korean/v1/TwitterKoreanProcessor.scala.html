39   /**
40    * Normalize Korean text. Uses KoreanNormalizer.normalize().
41    *
42    * @param text Input text
43    * @return Normalized Korean text
44    */
45   def normalize(text: CharSequence): CharSequence = KoreanNormalizer.normalize(text)
46 
47   /**
48    * Tokenize text into a sequence of token strings.
49    *
50    * @param text Input text, which can include any characters or spaces.
51    * @return A sequence of token strings.
52    */
53   def tokenizeToStrings(text: CharSequence): Seq[String] = {
54     tokenize(text).map(_.text.toString)
55   }
56 
57   /**
58    * Tokenize text into a sequence of normalized token strings.
59    *
60    * @param text Input text, which can include any characters or spaces.
61    * @return A sequence of normalized token strings.
62    */
63   def tokenizeToNormalizedStrings(text: CharSequence): Seq[String] = {
64     tokenizeWithNormalization(text).map(_.text.toString)
65   }
66 
67   /**
68    * Tokenize text into a sequence of KoreanTokens, which includes part-of-speech information and
69    * whether a token is an out-of-vocabulary term.
70    *
71    * @param text Input text.
72    * @return A sequence of KoreanTokens.
73    */
74   def tokenize(text: CharSequence): Seq[KoreanToken] = {
75     KoreanTokenizer.tokenize(text)
76   }
77 
78   /**
79    * Apply normalization before tokenization.
80    *
81    * @param text Input text.
82    * @return A sequence of normalized KoreanTokens.
83    */
84   def tokenizeWithNormalization(text: CharSequence): Seq[KoreanToken] = {
85     tokenize(normalize(text))
86   }
87 
88   /**
89    * Tokenize text into a sequence of KoreanSegments, which includes start offset, the length,
90    * and the full information of each token.
91    *
92    * This is useful for Lucene integration.
93    *
94    * @param text Input text.
95    * @return A sequence of KoreanSegments.
96    */
97   def tokenizeWithIndex(text: CharSequence): Seq[KoreanSegment] = {
98     val tokens: Seq[KoreanToken] = tokenize(text)
99 
100     val s: String = text.toString
101     // Match text with the tokenization results to get offset and length of each token
102     val (output, i) = tokens.foldLeft(List[KoreanSegment](), 0) {
103       case ((l: List[KoreanSegment], i: Int), token: KoreanToken) =>
104         val segStart = s.indexOf(token.text, i)
105         (KoreanSegment(segStart, token.text.length, token) :: l, segStart + token.text.length)
106     }
107     output.reverse
108   }
109 }
