30 /**
31  * Split input text into Korean Chunks (어절)
32  */
33 object KoreanChunker {
34   private val POS_PATTERNS = Map(
35     Korean -> """([가-힣]+)""".r.pattern,
36     Alpha -> """(\p{Alpha}+)""".r.pattern,
37     Number -> ("""(\$?\p{Digit}+"""
38       + """(,\p{Digit}{3})*([/~:\.-]\p{Digit}+)?"""
39       + """(천|만|억|조)*(%|원|달러|위안|옌|엔|유로|등|년|월|일|회|시간|시|분|초)?)""").r.pattern,
40     KoreanParticle -> """([ㄱ-ㅣ]+)""".r.pattern,
41     Punctuation -> """([\p{Punct}·…’]+)""".r.pattern,
42     URL -> Regex.VALID_URL,
43     Email -> """([\p{Alnum}\.\-_]+@[\p{Alnum}\.]+)""".r.pattern,
44     Hashtag -> Regex.VALID_HASHTAG,
45     ScreenName -> Regex.VALID_MENTION_OR_LIST,
46     CashTag -> Regex.VALID_CASHTAG,
47     Space -> """\s+""".r.pattern
48   )
49 
50   private val CHUNKING_ORDER = Seq(Space, URL, Email, ScreenName, Hashtag, CashTag, Number, Korean, KoreanParticle, Alpha, Punctuation)
51   private val SPACE_REGEX_DELIMITER_KEEP_SPACES = """((?<=\s+)|(?=\s+))"""
52 
53   protected[korean] def getChunks(input: String, keepSpace: Boolean = false): Seq[String] = {
54     chunk(input).map(_.text)
55   }
56 
57   private[this] case class ChunkMatch(start: Int, end: Int, text: String, pos: KoreanPos) {
58     def disjoint(that: ChunkMatch): Boolean = {
59       (that.start < this.start && that.end <= this.start) ||
60         (that.start >= this.end && that.end > this.end)
61     }
62   }
63 
64   /**
65    * Recursively call m.find() to find all the matches.
66    * Use tail-recursion optimization to avoid stack overflow.
67    *
68    * @param m input Matcher
69    * @param pos KoreanPos to attach
70    * @param matches ouput list of ChunkMatch
71    * @return list of ChunkMatches
72    */
73   @scala.annotation.tailrec
74   private[this] def findAllPatterns(m: Matcher, pos: KoreanPos, matches: List[ChunkMatch] = List()): List[ChunkMatch] = {
75     if (m.find()) {
76       findAllPatterns(m, pos, ChunkMatch(m.start, m.end, m.group(), pos) :: matches)
77     } else {
78       matches
79     }
80   }
81 
82   private[this] def splitChunks(text: String): List[ChunkMatch] = {
83     val chunks = text.split(SPACE_REGEX_DELIMITER_KEEP_SPACES).flatMap { s =>
84       CHUNKING_ORDER.foldLeft(List[ChunkMatch]()) {
85         (l, pos) =>
86           val m = POS_PATTERNS(pos).matcher(s)
87 
88           findAllPatterns(m, pos).filter(cm => l.forall(cm.disjoint)) ::: l
89       }
90     }.sortBy(cm => cm.start)
91 
92     fillInUnmatched(text, chunks, Foreign)
93   }
94 
95   /**
96    * Fill in unmatched segments with given pos
97    *
98    * @param text input text
99    * @param chunks matched chunks
100    * @param pos KoreanPos to attach to the unmatched chunk
101    * @return list of ChunkMatches
102    */
103   private[this] def fillInUnmatched(text: String,
104                                     chunks: Seq[ChunkMatch],
105                                     pos: KoreanPos.Value): List[ChunkMatch] = {
106 
107     // Add Foreign for unmatched parts
108     val (chunksWithForeign, prevEnd) = chunks.foldLeft((List[ChunkMatch](), 0)) {
109       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start == prevEnd =>
110         (cm :: l, cm.end)
111       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start > prevEnd =>
112         (cm :: ChunkMatch(prevEnd, cm.start, text.slice(prevEnd, cm.start), pos) :: l, cm.end)
113       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) =>
114         throw new IllegalStateException("Non-disjoint chunk matches found.")
115     }
116 
117     val output = if (prevEnd < text.length) {
118       ChunkMatch(prevEnd, text.length, text.slice(prevEnd, text.length), pos) :: chunksWithForeign
119     } else {
120       chunksWithForeign
121     }
122 
123     output.reverse
124   }
125 
126 
127   /**
128    * Get chunks by given pos.
129    *
130    * @param input input string
131    * @param pos one of supported KoreanPos's: URL, Email, ScreenName, Hashtag,
132    *            CashTag, Korean, KoreanParticle, Number, Alpha, Punctuation
133    * @return sequence of Korean chunk strings
134    */
135   def getChunksByPos(input: String, pos: KoreanPos): Seq[KoreanToken] = {
136     chunk(input).filter(_.pos == pos)
137   }
138 
139   /**
140    * Split input text into a sequnce of KoreanToken. A candidate for Korean parser
141    * gets tagged with KoreanPos.Korean.
142    *
143    * @param input input string
144    * @return sequence of KoreanTokens
145    */
146   def chunk(input: CharSequence): Seq[KoreanToken] = {
147     val splitRegex = SPACE_REGEX_DELIMITER_KEEP_SPACES
148     val s = input.toString
149 
150     val (l: List[KoreanToken], i: Int) = s.split(splitRegex).flatMap {
151       s => splitChunks(s)
152     }.foldLeft(List[KoreanToken](), 0) {
153       case ((l: List[KoreanToken], i: Int), m: ChunkMatch) =>
154         val segStart = s.indexOf(m.text, i)
155         (KoreanToken(m.text, m.pos, segStart, m.text.length) :: l, segStart + m.text.length)
156     }
157     l.reverse
158 
159   }
