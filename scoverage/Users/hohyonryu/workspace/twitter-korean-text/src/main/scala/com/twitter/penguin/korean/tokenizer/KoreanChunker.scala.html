28 /**
29  * Split input text into Korean Chunks (어절)
30  */
31 object KoreanChunker {
32   private val POS_PATTERNS = Map(
33     Korean -> """([가-힣]+)""".r.pattern,
34     Alpha -> """(\p{Alpha}+)""".r.pattern,
35     Number -> """(\p{Digit}+)""".r.pattern,
36     KoreanParticle -> """([ㄱ-ㅣ]+)""".r.pattern,
37     Punctuation -> """(\p{Punct}+)""".r.pattern,
38     URL -> Regex.VALID_URL,
39     Email -> """([\p{Alnum}\.\-_]+@[\p{Alnum}\.]+)""".r.pattern,
40     Hashtag -> Regex.VALID_HASHTAG,
41     ScreenName -> Regex.VALID_MENTION_OR_LIST,
42     CashTag -> Regex.VALID_CASHTAG,
43     Space -> """\s+""".r.pattern
44   )
45 
46   private val CHUNKING_ORDER = Seq(Space, URL, Email, ScreenName, Hashtag, CashTag, Korean, KoreanParticle, Number, Alpha, Punctuation)
47   private val SPACE_REGEX_DELIMITER_KEEP_SPACES = """((?<=\s+)|(?=\s+))"""
48   private val SPACE_REGEX_DELIMITER = """\s+"""
49 
50   protected[korean] def getChunks(input: String, keepSpace: Boolean = false): Seq[String] = {
51     chunk(input, keepSpace).map(_.text)
52   }
53 
54 
55   private[this] case class ChunkMatch(start: Int, end: Int, text: String, pos: KoreanPos) {
56     def disjoint(that: ChunkMatch): Boolean = {
57       (that.start < this.start && that.end <= this.start) ||
58           (that.start >= this.end && that.end > this.end)
59     }
60   }
61 
62   /**
63    * Recursively call m.find() to find all the matches.
64    * Use tail-recursion optimization to avoid stack overflow.
65    *
66    * @param m input Matcher
67    * @param pos KoreanPos to attach
68    * @param matches ouput list of ChunkMatch
69    * @return list of ChunkMatches
70    */
71   @scala.annotation.tailrec
72   private[this] def findAllPatterns(m: Matcher, pos: KoreanPos, matches: List[ChunkMatch] = List()): List[ChunkMatch] = {
73     if (m.find()) {
74       findAllPatterns(m, pos, ChunkMatch(m.start, m.end, m.group(), pos) :: matches)
75     } else {
76       matches
77     }
78   }
79 
80   private[this] def splitChunks(text: String, keepSpace: Boolean = false): List[ChunkMatch] = {
81     val splitRegex = if (keepSpace) SPACE_REGEX_DELIMITER_KEEP_SPACES else SPACE_REGEX_DELIMITER
82     val chunks = text.split(splitRegex).flatMap { s =>
83       CHUNKING_ORDER.foldLeft(List[ChunkMatch]()) {
84         (l, pos) =>
85           val m = POS_PATTERNS(pos).matcher(s)
86 
87           findAllPatterns(m, pos).filter(cm => l.forall(cm.disjoint)) ::: l
88       }
89     }.sortBy(cm => cm.start)
90 
91     fillInUnmatched(text, chunks, Foreign)
92   }
93 
94   /**
95    * Fill in unmatched segments with given pos
96    *
97    * @param text input text
98    * @param chunks matched chunks
99    * @param pos KoreanPos to attach to the unmatched chunk
100    * @return list of ChunkMatches
101    */
102   private[this] def fillInUnmatched(text: String,
103                                     chunks: Seq[ChunkMatch],
104                                     pos: KoreanPos.Value): List[ChunkMatch] = {
105 
106     // Add Foreign for unmatched parts
107     val (chunksWithForeign, prevEnd) = chunks.foldLeft((List[ChunkMatch](), 0)) {
108       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start == prevEnd =>
109         (cm :: l, cm.end)
110       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start > prevEnd =>
111         (cm :: ChunkMatch(prevEnd, cm.start, text.slice(prevEnd, cm.start), pos) :: l, cm.end)
112       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) =>
113         throw new IllegalStateException("Non-disjoint chunk matches found.")
114     }
115 
116     val output = if (prevEnd < text.length) {
117       ChunkMatch(prevEnd, text.length, text.slice(prevEnd, text.length), pos) :: chunksWithForeign
118     } else {
119       chunksWithForeign
120     }
121 
122     output.reverse
123   }
124 
125 
126   /**
127    * Get chunks by given pos.
128    *
129    * @param input input string
130    * @param pos one of supported KoreanPos's: URL, Email, ScreenName, Hashtag,
131    *            CashTag, Korean, KoreanParticle, Number, Alpha, Punctuation
132    * @return sequence of Korean chunk strings
133    */
134   def getChunksByPos(input: String, pos: KoreanPos): Seq[String] = {
135     chunk(input).filter(_.pos == pos).map(_.text)
136   }
137 
138   /**
139    * Split input text into a sequnce of KoreanToken. A candidate for Korean parser
140    * gets tagged with KoreanPos.Korean.
141    *
142    * @param input input string
143    * @return sequence of KoreanTokens
144    */
145   def chunk(input: CharSequence, keepSpace: Boolean = false): Seq[KoreanToken] = {
146     val splitRegex = if (keepSpace) SPACE_REGEX_DELIMITER_KEEP_SPACES else SPACE_REGEX_DELIMITER
147     input.toString
148         .split(splitRegex)
149         .flatMap(s => splitChunks(s, keepSpace))
150         .map(m => KoreanToken(m.text, m.pos))
151   }
152 }
