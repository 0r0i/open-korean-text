36   protected[korean] def isJosaAttachable(prevChar: Char, headChar: Char): Boolean = {
37     (hasCoda(prevChar) && !JOSA_HEAD_FOR_NO_CODA.contains(headChar)) ||
38         (!hasCoda(prevChar) && !JOSA_HEAD_FOR_CODA.contains(headChar))

41   protected[korean] def isName(chunk: CharSequence): Boolean = {
42     if (chunk.length() != 3) return false
43     nameDictionay('family_name).contains(chunk.charAt(0).toString) &&
44         nameDictionay('given_name).contains(chunk.subSequence(1, 3).toString)
45   }

50   protected[korean] def isKoreanNumber(chunk: CharSequence): Boolean =
51     (0 to chunk.length() - 1).foldLeft(true) {
52       case (output, i) if i < chunk.length() - 1 => output && NUMBER_CHARS.contains(chunk.charAt(i).toInt)
53       case (output, i) => output && NUMBER_LAST_CHARS.contains(chunk.charAt(i).toInt)
54     }

57   /**
58    * Collapse all the one-char nouns into one unknown noun
59    *
60    * @param posNodes sequence of KoreanTokens
61    * @return sequence of collapsed KoreanTokens
62    */
63   protected[korean] def collapseNouns(posNodes: Seq[KoreanToken]): Seq[KoreanToken] = {
64     val (nodes, collapsing) = posNodes.foldLeft((List[KoreanToken](), false)) {
65       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken)
66         if p.pos == Noun && p.text.length == 1 && collapsing =>
67         (KoreanToken(pl.head.text + p.text, Noun, unknown = true) :: pl.tail, true)
68       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken)
69         if p.pos == Noun && p.text.length == 1 && !collapsing =>
70         (p :: pl, true)
71       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken) =>
72         (p :: pl, false)
73     }
74     nodes.reverse.toSeq
75   }
