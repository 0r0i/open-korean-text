36   protected[korean] def isJosaAttachable(prevChar: Char, headChar: Char): Boolean = {
37     (hasCoda(prevChar) && !JOSA_HEAD_FOR_NO_CODA.contains(headChar)) ||
38         (!hasCoda(prevChar) && !JOSA_HEAD_FOR_CODA.contains(headChar))
39   }

41   protected[korean] def isName(chunk: CharSequence): Boolean = {
42     if (nameDictionay('full_name).contains(chunk) ||
43         nameDictionay('given_name).contains(chunk)) return true
44 
45     if (chunk.length() != 3) return false
46     nameDictionay('family_name).contains(chunk.charAt(0).toString) &&
47         nameDictionay('given_name).contains(chunk.subSequence(1, 3).toString)
48   }

50   private val NUMBER_CHARS = "일이삼사오육칠팔구천백십해경조억만".map(_.toInt).toSet
51   private val NUMBER_LAST_CHARS = "일이삼사오육칠팔구천백십해경조억만원배분초".map(_.toInt).toSet

53   protected[korean] def isKoreanNumber(chunk: CharSequence): Boolean =
54     (0 to chunk.length() - 1).foldLeft(true) {
55       case (output, i) if i < chunk.length() - 1 => output && NUMBER_CHARS.contains(chunk.charAt(i).toInt)
56       case (output, i) => output && NUMBER_LAST_CHARS.contains(chunk.charAt(i).toInt)
57     }

59   /**
60    * Check if this chunk is an 'ㅇ' omitted variation of a noun (우혀니 -> 우현, 우현이, 빠순이 -> 빠순, 빠순이)
61    *
62    * @param chunk input chunk
63    * @return true if the chunk is an 'ㅇ' omitted variation
64    */
65   protected[korean] def isKoreanNameVariation(chunk: CharSequence): Boolean = {
66     val nounDict = koreanDictionary(Noun)
67 
68     val s = chunk.toString
69     if (isName(s)) return true
70     if (s.length < 3 || s.length > 5) return false
71 
72     val decomposed = s.map { c: Char => decomposeHangul(c)}
73     val lastChar = decomposed.last
74     if (!Hangul.CODA_MAP.contains(lastChar.onset)) return false
75     if (lastChar.onset == 'ㅇ' || lastChar.vowel != 'ㅣ' || lastChar.coda != ' ') return false
76     if (decomposed.init.last.coda != ' ') return false
77 
78     // Recover missing 'ㅇ' (우혀니 -> 우현, 우현이, 빠순이 -> 빠순, 빠순이)
79     val recovered = decomposed.zipWithIndex.map {
80       case (hc: HangulChar, i: Int) if i == s.length - 1 => '이'
81       case (hc: HangulChar, i: Int) if i == s.length - 2 =>
82         composeHangul(HangulChar(hc.onset, hc.vowel, decomposed.last.onset))
83       case (hc: HangulChar, i: Int) => composeHangul(hc)
84     }.mkString("")
85 
86     Seq(recovered, recovered.init).exists(isName)
87   }

89   /**
90    * Collapse all the one-char nouns into one unknown noun
91    *
92    * @param posNodes sequence of KoreanTokens
93    * @return sequence of collapsed KoreanTokens
94    */
95   protected[korean] def collapseNouns(posNodes: Seq[KoreanToken]): Seq[KoreanToken] = {
96     val (nodes, collapsing) = posNodes.foldLeft((List[KoreanToken](), false)) {
97       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken)
98         if p.pos == Noun && p.text.length == 1 && collapsing =>
99         (KoreanToken(pl.head.text + p.text, Noun, unknown = true) :: pl.tail, true)
100       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken)
101         if p.pos == Noun && p.text.length == 1 && !collapsing =>
102         (p :: pl, true)
103       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken) =>
104         (p :: pl, false)
105     }
106     nodes.reverse.toSeq
107   }
