27 /**
28  * Create Korean Parsing goldenset from the goldenset resource that contains goldenset chunks.
29  * The first argument is a gzipped output file.
30  */
31 object CreateParsingGoldenset {
32 
33   case class ParsingExample(text: String, parse: Seq[KoreanToken])
34 
35   def main(args: Array[String]) {
36     System.err.println("Reading the goldenset..")
37 
38     val parsedPairs = readFileByLineFromResources("example_chunks.txt").flatMap {
39       case line if line.length > 0 =>
40         val chunk = line.trim
41         val parsed = tokenize(chunk)
42         Some(ParsingExample(chunk, parsed))
43       case line => None
44     }.toSet
45 
46 
47     val outputFile: String = "src/test/resources/com/twitter/penguin/korean/v1/util/current_parsing.txt"
48 
49     System.err.println("Writing the new goldenset to " + outputFile)
50 
51     val out = new FileOutputStream(outputFile)
52     parsedPairs.foreach {
53       p =>
54         out.write(p.text.getBytes)
55         out.write("\t".getBytes)
56         out.write(p.parse.mkString("/").getBytes)
57         out.write("\n".getBytes)
58     }
59     out.close()
60 
61     System.err.println("Testing the new goldenset " + outputFile)
62   }
63 }
