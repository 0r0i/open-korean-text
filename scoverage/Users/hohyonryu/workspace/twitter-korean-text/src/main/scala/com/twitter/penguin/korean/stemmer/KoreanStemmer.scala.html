7 /**
8  * Stems Adjectives and Verbs: 새로운 스테밍을 추가했었다. -> 새롭다 + 스테밍 + 을 + 추가하다
9  */
10 object KoreanStemmer {
11   private val Endings = Set(Eomi, PreEomi)
12   private val Predicates = Set(Verb, Adjective)
13 
14   private val EndingsForNouns = Set("하다", "되다", "없다")
15 
16   /**
17    * Removes Ending tokens recovering the root form of predicates
18    *
19    * @param tokens A sequence of tokens
20    * @return A sequence of collapsed Korean tokens
21    */
22   def stem(tokens: Seq[KoreanToken]): Seq[KoreanToken] = {
23     if (!tokens.exists(t => t.pos == Verb || t.pos == Adjective)) {
24       return tokens
25     }
26 
27     val stemmed = tokens.foldLeft(List[KoreanToken]()) {
28       case (l: List[KoreanToken], token: KoreanToken) if Endings.contains(token.pos) =>
29         if (Predicates.contains(l.head.pos)) {
30           val prevToken = l.head
31           KoreanToken(
32             prevToken.text,
33             prevToken.pos, prevToken.offset, prevToken.length + token.length, prevToken.unknown
34           ) :: l.tail
35         } else {
36           l
37         }
38       case (l: List[KoreanToken], token: KoreanToken) if Predicates.contains(token.pos) =>
39         val text = predicateStems(token.pos)(token.text)
40         KoreanToken(
41           text,
42           token.pos, token.offset, token.length, token.unknown
43         ) :: l
44       case (l: List[KoreanToken], token: KoreanToken) => token :: l
45     }.reverse
46 
47     def validNounHeading(token: KoreanToken): Boolean = {
48       val heading = token.text.take(token.text.length - 2)
49 
50       val validLength = token.text.length > 2
51       val validPos = token.pos == Verb
52       val validEndings = EndingsForNouns.contains(token.text.takeRight(2))
53       val validNouns = koreanDictionary(Noun).contains(heading)
54 
55       validLength && validPos && validEndings && validNouns
56     }
57 
58     stemmed.flatMap {
59       case token if validNounHeading(token) =>
60         val heading = token.text.take(token.text.length - 2)
61         val ending = token.text.takeRight(2)
62 
63         Seq(
64           KoreanToken(heading, Noun, token.offset, heading.length),
65           KoreanToken(ending, token.pos, token.offset + heading.length, token.length - heading.length)
66         )
67       case token => Seq(token)
68     }
69   }
70 }
