11 object KoreanStemmer {
12   private val Endings = Set(Eomi, PreEomi)
13   private val Predicates = Set(Verb, Adjective)

15   private val EndingsForNouns = Set("하다", "되다", "없다")

19   /**
20    * Removes Ending tokens recovering the root form of predicates
21    *
22    * @param tokens A sequence of tokens
23    * @return A sequence of collapsed Korean tokens ([새, 로, 운, 사람] -> [[새, 로, 운].], 사람)
24    */
25   def stemPredicatesCore(tokens: Seq[KoreanToken]): Seq[Seq[KoreanToken]] = {
26     if (!tokens.exists(t => t.pos == Verb || t.pos == Adjective)) {
27       return tokens.map(t => Seq(t));
28     }
29 
30     val stemmed = tokens.map {
31       case token: KoreanToken if Endings.contains(token.pos) => None
32       case token: KoreanToken if Predicates.contains(token.pos) =>
33         Some(
34           KoreanToken(
35             predicateStems(token.pos)(token.text),
36 
37             token.pos, token.unknown
38           )
39         )
40       case token => Some(token)
41     }
42 
43     def validNounHeading(token: KoreanToken): Boolean = {
44       val heading = token.text.take(token.text.length - 2)
45 
46       val validLength = token.text.length > 2
47       val validPos = token.pos == Verb
48       val validEndings = EndingsForNouns.contains(token.text.takeRight(2))
49       val validNouns = koreanDictionary(Noun).contains(heading)
50 
51       validLength && validPos && validEndings && validNouns
52     }
53 
54     stemmed.map {
55       case Some(token) if validNounHeading(token) =>
56         val heading = token.text.take(token.text.length - 2)
57         val ending = token.text.takeRight(2)
58 
59         Seq(
60           KoreanToken(heading, Noun),
61           KoreanToken(ending, token.pos)
62         )
63       case Some(token) => Seq(token)
64       case None => Seq()
65     }
66   }

68   /**

70    *
71    * @param tokens A sequence of tokens
72    * @return A list of Korean tokens
73    */
74   def stemPredicates(tokens: Seq[KoreanToken]): Seq[KoreanToken] = {
75     stemPredicatesCore(tokens).flatMap(c => c)
76   }

78   /**
79    * Stem tokens
80    *
81    * @param text Input text
82    * @return StemmedTextWithTokens
83    */
84   def stem(text: CharSequence): StemmedTextWithTokens = {
85     val tokenized = KoreanTokenizer.tokenize(text)
86 
87     val stemmed = stemPredicatesCore(tokenized)
88     val mapped = tokenized.zip(stemmed)
89 
90     val s = text.toString
91     // transform the original text to the stemmed text
92     val (sb, i) = mapped.foldLeft(new StringBuilder(), 0) {
93       case ((sb: StringBuilder, i: Int), (token: KoreanToken, stemmedTokens: Seq[KoreanToken])) =>
94         val segStart = s.indexOf(token.text, i)
95         sb.append(s.substring(i, segStart))
96         sb.append(stemmedTokens.map(_.text).mkString(""))
97         (sb, segStart + token.text.length)
98     }
99     sb.append(s.substring(i, text.length))
100 
101     StemmedTextWithTokens(sb, stemmed.flatten)
102   }
