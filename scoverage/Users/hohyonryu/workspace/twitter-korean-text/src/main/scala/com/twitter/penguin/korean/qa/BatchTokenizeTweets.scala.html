29 /**
30  * Batch tokenize a file of Korean tweets.
31  *
32  * Program arguments example = "~/korean_201407_100"
33  */
34 object BatchTokenizeTweets {
35 
36   case class ParseTime(time: Long, chunk: String)
37 
38   private val LOG = Logger.getLogger(getClass.getSimpleName)
39   private val VERBOSE = true
40   private val NON_NOUNS = Set(KoreanPos.Adjective, KoreanPos.Adverb, KoreanPos.Verb)
41 
42   def main(args: Array[String]) {
43     if (args.length != 1) {
44       println("The first arg should be an input file of Korean tweets.")
45       return
46     }
47     val parseTimesAll = Source.fromFile(args(0)).getLines().foldLeft(List[ParseTime]()) {
48       case (l: List[ParseTime], line: String) =>
49         val t0 = System.currentTimeMillis()
50         val parsed = TwitterKoreanProcessor.tokenize(line)
51         val t1 = System.currentTimeMillis()
52 
53         if (VERBOSE) {
54           println(parsed.map(t => t.text + "/" + t.pos).mkString(" "))
55         }
56         ParseTime(t1 - t0, line.trim) :: l
57     }
58 
59     val loadingTime = parseTimesAll.last
60 
61     LOG.log(Level.INFO, "The first one \"%s\" took %d ms including the loading time.".format(loadingTime.chunk, loadingTime.time))
62 
63     val parseTimes = parseTimesAll.init
64 
65     val averageTweetLength = parseTimes.map(_.chunk.length).sum.toDouble / parseTimes.size
66 
67     val averageTime = parseTimes.map(_.time).sum.toDouble / parseTimes.size
68     val maxItem = parseTimes.maxBy(_.time)
69 
70     LOG.log(Level.INFO, ("Parsed %d items. \n" +
71         "       Total time: %d s \n" +
72         "       Average tweet length: %.2f chars \n" +
73         "       Average time per tweet: %.2f ms \n" +
74         "       Max time: %d ms, %s\n" +
75         "       Parsed: %s"
76         ).format(
77           parseTimes.size,
78           parseTimes.map(_.time).sum / 1000,
79           averageTweetLength,
80           averageTime,
81           maxItem.time,
82           maxItem.chunk,
83           TwitterKoreanProcessor.tokenize(maxItem.chunk).map {
84             case t if t.unknown => t.text.toString + t.pos + "*"
85             case t => t.text + t.pos.toString
86           }.mkString(" ")
87         ))
88   }
89 
90   private def parseToString(parsed: Seq[KoreanToken]): String = {
91     parsed.map {
92       case t if t.unknown => t.text.toString + t.pos + "*"
93       case t => t.text + t.pos.toString
94     }.mkString(" ")
95   }
96 }
