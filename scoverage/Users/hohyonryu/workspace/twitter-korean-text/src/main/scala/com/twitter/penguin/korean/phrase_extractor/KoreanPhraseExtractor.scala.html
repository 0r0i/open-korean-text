29    * 0 for optional, 1 for required
30    * * for optional repeatable, + for required repeatable
31    *
32    * Substantive: 체언 (초거대기업의)
33    * Predicate: 용언 (하였었습니다, 개예뻤었다)
34    * Modifier: 수식언 (모르는 할수도있는 보이기도하는 예뻐 예쁜 완전 레알 초인간적인 잘 잘한)

36    * Functional: 관계언 (조사)
37    *
38    * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
39    * V Verb: 동사 (하, 먹, 자, 차)
40    * J Adjective: 형용사 (예쁘다, 크다, 작다)
41    * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
42    * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
43    * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
44    *
45    * C Conjunction: 접속사
46    *
47    * j SubstantiveJosa: 조사 (의, 에, 에서)

69     def trimNonNouns: Seq[KoreanPhrase] = {

86           KoreanPhrase(phrase.tokens.reverse.dropWhile(_.pos == Space).reverse, phrase.pos)

100   private def isProperPhraseChunk(phraseChunk: KoreanPhraseChunk): Boolean = {
101     def notEndingInNonPhraseSuffix: Boolean = {
102       val lastToken = phraseChunk.last.tokens.last
103       !(lastToken.pos == Suffix && lastToken.text == "적")
104     }
105 
106     def isRightLength: Boolean = {
107       val phraseChunkWithoutSpaces: Seq[KoreanPhrase] = phraseChunk.filter(_.pos != Space)
108 
109       def checkMaxLength: Boolean = {
110         phraseChunkWithoutSpaces.length <= MaxPhrasesPerPhraseChunk &&
111             phraseChunkWithoutSpaces.map(_.getTextLength).sum <= MaxCharsPerPhraseChunkWithoutSpaces
112       }
113 
114       def checkMinLength: Boolean = {
115         phraseChunkWithoutSpaces.length >= MinPhrasesPerPhraseChunk ||
116             (phraseChunkWithoutSpaces.length < MinPhrasesPerPhraseChunk &&
117                 phraseChunkWithoutSpaces.map(_.getTextLength).sum >= MinCharsPerPhraseChunkWithoutSpaces)
118       }
119 
120       def checkMinLengthPerToken: Boolean = {
121         phraseChunkWithoutSpaces.exists(_.getTextLength > 1)
122       }
123 
124       checkMaxLength && checkMinLength && checkMinLengthPerToken
125     }
126 
127     isRightLength && notEndingInNonPhraseSuffix
128   }

146         case nt: KoreanPosTrie if nt == selfNode => curTrie

152     def getInit(phraseBuffer: PhraseBuffer): List[KoreanPhrase] = {
153       if (phraseBuffer.phrases.isEmpty) {
154         List()
155       } else {
156         phraseBuffer.phrases.init
157       }
158     }
159

161       case (output, token) if output.curTrie.exists(_.curPos == token.pos) =>
162         val (ct, nt) = getTries(token, output.curTrie)
163

170           PhraseBuffer(
171             List(KoreanPhrase(List(token), ct.ending.getOrElse(Noun))),
172             nt, ct.ending
173           )
174         } else {

177                 KoreanPhrase(output.phrases.last.tokens :+ token, ct.ending.getOrElse(Noun)),
178             nt, ct.ending
179           )
180         }
181       case (output, token) if CollapseTrie.exists(_.curPos == token.pos) =>
182         val (ct, nt) = getTries(token, CollapseTrie)
183 
184         PhraseBuffer(
185           output.phrases :+ KoreanPhrase(List(token), ct.ending.getOrElse(Noun)),
186           nt, ct.ending
187         )
188       case (output, token) =>
189         PhraseBuffer(
190           output.phrases :+ KoreanPhrase(List(token), token.pos),
191           output.curTrie, output.ending
192         )
193     }.phrases

196   protected def getCandidatePhraseChunks(phrases: KoreanPhraseChunk): Seq[KoreanPhraseChunk] = {
197     def isNonNounPhraseCandidate(phrase: KoreanPhrase): Boolean = {
198       val trimmed = trimPhrase(phrase)
199 
200       // 하는, 할인된, 할인될
201       def isModifyingPredicate: Boolean =
202         (trimmed.pos == Verb || trimmed.pos == Adjective) &&
203             ModifyingPredicateEndings.contains(Hangul.decomposeHangul(trimmed.tokens.last.text.last).coda)
204 
205       // 과, 와, 의, 그리고
206       def isConjuction: Boolean =
207         trimmed.pos == Josa && ConjunctionJosa.contains(trimmed.tokens.last.text) ||
208             trimmed.pos == Conjunction
209 
210       def isAlphaNumeric: Boolean =
211         trimmed.pos == Alpha || trimmed.pos == Number
212 
213       isAlphaNumeric || isModifyingPredicate || isConjuction
214     }

216     def collapseNounPhrases(phrases: KoreanPhraseChunk): KoreanPhraseChunk = {
217       val (output, buffer) = phrases.foldLeft((Seq[KoreanPhrase](), Seq[KoreanPhrase]())) {
218         case ((output, buffer), phrase) if phrase.pos == Noun => (output, buffer :+ phrase)
219         case ((output, buffer), phrase) =>
220           val tempPhrases = if (buffer.length > 0) {
221             Seq(KoreanPhrase(buffer.flatMap(_.tokens)), phrase)
222           } else {
223             Seq(phrase)
224           }
225           (output ++ tempPhrases, Seq[KoreanPhrase]())
226       }
227       if (buffer.length > 0) output :+ KoreanPhrase(buffer.flatMap(_.tokens)) else output
228     }

230     def collapsePhrases(phrases: KoreanPhraseChunk): Seq[KoreanPhraseChunk] = {
231       def addPhraseToBuffer(phrase: KoreanPhrase, buffer: Seq[KoreanPhraseChunk]): Seq[KoreanPhraseChunk] = {
232         buffer.map(b => b :+ phrase)
233       }
234 
235       def newBuffer: Seq[Seq[KoreanPhrase]] = Seq(Seq[KoreanPhrase]())
236 
237       val (output, buffer) = phrases.foldLeft(
238         (Seq[KoreanPhraseChunk](), newBuffer)
239       ) {
240         case ((output, buffer), phrase) if PhraseTokens.contains(phrase.pos) =>
241           (output, addPhraseToBuffer(phrase, buffer))
242         case ((output, buffer), phrase) if buffer.length > 0 && isNonNounPhraseCandidate(phrase) =>
243           (output ++ buffer, addPhraseToBuffer(phrase, buffer :+ Seq[KoreanPhrase]()))
244         case ((output, buffer), phrase) if isNonNounPhraseCandidate(phrase) =>
245           (output, addPhraseToBuffer(phrase, buffer))
246         case ((output, buffer), phrase) if buffer.length > 0 =>
247           (output ++ buffer, newBuffer)
248         case ((output, buffer), phrase) => (output, buffer)
249       }
250       if (buffer.length > 0) output ++ buffer else output
251     }

253     def getSingleTokenNouns: Seq[KoreanPhraseChunk] = {
254       phrases.filter {
255         phrase =>
256           val trimmed = trimPhrase(phrase)
257           phrase.pos == Noun &&
258               (trimmed.getTextLength >= MinCharsPerPhraseChunkWithoutSpaces ||
259                   trimmed.tokens.length >= MinPhrasesPerPhraseChunk)
260       }.map(phrase => Seq(trimPhrase(phrase)))
261     }
262 
263     val nounPhrases: KoreanPhraseChunk = collapseNounPhrases(phrases)
264     val phraseCollapsed = collapsePhrases(nounPhrases)
265 
266     (phraseCollapsed.map(trimPhraseChunk) ++ getSingleTokenNouns).distinct
267   }

270    * Find suitable phrases
271    *
272    * @param tokens A sequence of tokens
273    * @return A list of KoreanPhrase
274    */
275   def extractPhrases(tokens: Seq[KoreanToken]): Seq[KoreanPhrase] = {
276     val collapsed = collapsePos(tokens)
277     val candidates = getCandidatePhraseChunks(collapsed)
278 
279     val permutatedCandidates = candidates.flatMap {
280       case phrases if phrases.length > MinPhrasesPerPhraseChunk =>
281         (0 to phrases.length - MinPhrasesPerPhraseChunk).map {
282           i => trimPhraseChunk(phrases.slice(i, phrases.length))
283         }
284       case phrases => Seq(phrases)
285     }.filter { phraseChunk: KoreanPhraseChunk => isProperPhraseChunk(phraseChunk)
286     }.distinct
287 
288     permutatedCandidates.map {
289       phraseChunk: KoreanPhraseChunk => KoreanPhrase(trimPhraseChunk(phraseChunk).flatMap(_.tokens))
290     }
291   }
