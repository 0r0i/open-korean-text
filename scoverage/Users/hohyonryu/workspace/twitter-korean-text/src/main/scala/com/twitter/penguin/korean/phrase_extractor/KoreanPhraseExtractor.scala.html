162     def getInit(phraseBuffer: PhraseBuffer): List[KoreanPhrase] = {
163       if (phraseBuffer.phrases.isEmpty) {
164         List()
165       } else {
166         phraseBuffer.phrases.init
167       }
168     }

180         } else {

183               KoreanPhrase(output.phrases.last.tokens :+ token, ct.ending.getOrElse(Noun)),
184             nt, ct.ending

186         }

192           output.phrases :+ KoreanPhrase(List(token), ct.ending.getOrElse(Noun)),

196         // Add a single word

198           output.phrases :+ KoreanPhrase(List(token), token.pos),

224     def isNonNounPhraseCandidate(phrase: KoreanPhrase): Boolean = {
225       val trimmed = trimPhrase(phrase)

227       // 하는, 할인된, 할인될, exclude: 하지만
228       def isModifyingPredicate: Boolean = {
229         val lastChar: Char = trimmed.tokens.last.text.last
230         (trimmed.pos == Verb || trimmed.pos == Adjective) &&
231           ModifyingPredicateEndings.contains(Hangul.decomposeHangul(lastChar).coda) &&
232           !ModifyingPredicateExceptions.contains(lastChar)
233       }
234 
235 
236       // 과, 와, 의
237       def isConjuction: Boolean =
238         trimmed.pos == Josa && ConjunctionJosa.contains(trimmed.tokens.last.text)

243       isAlphaNumeric || isModifyingPredicate || isConjuction

246     def collapseNounPhrases(phrases: KoreanPhraseChunk): KoreanPhraseChunk = {
247       val (output, buffer) = phrases.foldLeft((Seq[KoreanPhrase](), Seq[KoreanPhrase]())) {

253           } else {
254             Seq(phrase)
255           }

261     def collapsePhrases(phrases: KoreanPhraseChunk): Seq[KoreanPhraseChunk] = {
262       def addPhraseToBuffer(phrase: KoreanPhrase, buffer: Seq[KoreanPhraseChunk]): Seq[KoreanPhraseChunk] = {
263         buffer.map(b => b :+ phrase)
264       }
265 
266       def newBuffer: Seq[Seq[KoreanPhrase]] = Seq(Seq[KoreanPhrase]())
267 
268       val (output, buffer) = phrases.foldLeft((Seq[KoreanPhraseChunk](), newBuffer)) {
269         case ((output, buffer), phrase) if PhraseTokens.contains(phrase.pos) && isNotSpam(phrase) =>
270           val bufferWithThisPhrase = addPhraseToBuffer(phrase, buffer)
271           if (phrase.pos == Noun || phrase.pos == ProperNoun) {
272             (output ++ bufferWithThisPhrase, bufferWithThisPhrase)
273           } else {
274             (output, bufferWithThisPhrase)
275           }
276         case ((output, buffer), phrase) if isNonNounPhraseCandidate(phrase) =>
277           (output, addPhraseToBuffer(phrase, buffer))
278         case ((output, buffer), phrase) =>
279           (output ++ buffer, newBuffer)
280       }
281       if (buffer.length > 0) output ++ buffer else output
282     }

308   def extractPhrases(tokens: Seq[KoreanToken],
309                      filterSpam: Boolean = false,
310                      addHashtags: Boolean = true): Seq[KoreanPhrase] = {
311     val hashtags = tokens.flatMap {
312       case t: KoreanToken if t.pos == KoreanPos.Hashtag => Some(KoreanPhrase(Seq(t), KoreanPos.Hashtag))
313       case t: KoreanToken if t.pos == KoreanPos.CashTag => Some(KoreanPhrase(Seq(t), KoreanPos.CashTag))
314       case _ => None
315     }
316 
317     val collapsed = collapsePos(tokens)
318     val candidates = getCandidatePhraseChunks(collapsed, filterSpam)
319     val permutatedCandidates = permutateCadidates(candidates)
320 
321     val phrases = permutatedCandidates.map {
322       phraseChunk: KoreanPhraseChunk => KoreanPhrase(trimPhraseChunk(phraseChunk).flatMap(_.tokens))
323     }
324 
325     if (addHashtags) {
326       phrases ++ hashtags
327     } else {
328       phrases
329     }
330   }

333   private def permutateCadidates(candidates: Seq[KoreanPhraseChunk]): Seq[KoreanPhraseChunk] = {
334     val permutated = candidates.flatMap {
335       case phrases if phrases.length > MinPhrasesPerPhraseChunk =>
336         (0 to phrases.length - MinPhrasesPerPhraseChunk).map {
337           i => trimPhraseChunk(phrases.slice(i, phrases.length))
338         }
339       case phrases => Seq(phrases)
340     }.filter { phraseChunk: KoreanPhraseChunk => isProperPhraseChunk(phraseChunk)
341     }
342     distinctPhrases(permutated)
343   }
