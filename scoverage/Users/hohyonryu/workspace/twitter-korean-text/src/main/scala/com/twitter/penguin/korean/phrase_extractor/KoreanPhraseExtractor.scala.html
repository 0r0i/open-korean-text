102   private def isProperPhraseChunk(phraseChunk: KoreanPhraseChunk): Boolean = {
103     def notEndingInNonPhraseSuffix: Boolean = {
104       val lastToken = phraseChunk.last.tokens.last
105       !(lastToken.pos == Suffix && lastToken.text == "적")
106     }
107 
108     def isRightLength: Boolean = {
109       val phraseChunkWithoutSpaces: Seq[KoreanPhrase] = phraseChunk.filter(_.pos != Space)
110 
111       def checkMaxLength: Boolean = {
112         phraseChunkWithoutSpaces.length <= MaxPhrasesPerPhraseChunk &&
113             phraseChunkWithoutSpaces.map(_.getTextLength).sum <= MaxCharsPerPhraseChunkWithoutSpaces
114       }
115 
116       def checkMinLength: Boolean = {
117         phraseChunkWithoutSpaces.length >= MinPhrasesPerPhraseChunk ||
118             (phraseChunkWithoutSpaces.length < MinPhrasesPerPhraseChunk &&
119                 phraseChunkWithoutSpaces.map(_.getTextLength).sum >= MinCharsPerPhraseChunkWithoutSpaces)
120       }
121 
122       def checkMinLengthPerToken: Boolean = {
123         phraseChunkWithoutSpaces.exists(_.getTextLength > 1)
124       }

126       checkMaxLength && checkMinLength && checkMinLengthPerToken
127     }

198   protected def getCandidatePhraseChunks(phrases: KoreanPhraseChunk,
199                                          filterSpam: Boolean = false): Seq[KoreanPhraseChunk] = {
200     def isNotSpam(phrase: KoreanPhrase): Boolean =

205     def isNonNounPhraseCandidate(phrase: KoreanPhrase): Boolean = {
206       val trimmed = trimPhrase(phrase)
207 
208       // 하는, 할인된, 할인될, exclude: 하지만
209       def isModifyingPredicate: Boolean = {
210         val lastChar: Char = trimmed.tokens.last.text.last
211         (trimmed.pos == Verb || trimmed.pos == Adjective) &&
212             ModifyingPredicateEndings.contains(Hangul.decomposeHangul(lastChar).coda) &&
213             !ModifyingPredicateExceptions.contains(lastChar)
214       }
215 
216 
217       // 과, 와, 의
218       def isConjuction: Boolean =
219         trimmed.pos == Josa && ConjunctionJosa.contains(trimmed.tokens.last.text)
220 
221       def isAlphaNumeric: Boolean =
222         trimmed.pos == Alpha || trimmed.pos == Number
223 
224       isAlphaNumeric || isModifyingPredicate || isConjuction
225     }

228       val (output, buffer) = phrases.foldLeft((Seq[KoreanPhrase](), Seq[KoreanPhrase]())) {
229         case ((output, buffer), phrase) if phrase.pos == Noun => (output, buffer :+ phrase)
230         case ((output, buffer), phrase) =>

232             Seq(KoreanPhrase(buffer.flatMap(_.tokens)), phrase)
233           } else {
234             Seq(phrase)
235           }

241     def collapsePhrases(phrases: KoreanPhraseChunk): Seq[KoreanPhraseChunk] = {
242       def addPhraseToBuffer(phrase: KoreanPhrase, buffer: Seq[KoreanPhraseChunk]): Seq[KoreanPhraseChunk] = {
243         buffer.map(b => b :+ phrase)
244       }
245 
246       def newBuffer: Seq[Seq[KoreanPhrase]] = Seq(Seq[KoreanPhrase]())
247 
248       val (output, buffer) = phrases.foldLeft(
249         (Seq[KoreanPhraseChunk](), newBuffer)
250       ) {
251         case ((output, buffer), phrase) if PhraseTokens.contains(phrase.pos) && isNotSpam(phrase) =>
252           (output, addPhraseToBuffer(phrase, buffer))
253         case ((output, buffer), phrase) if buffer.length > 0 && isNonNounPhraseCandidate(phrase) =>
254           (output ++ buffer, addPhraseToBuffer(phrase, buffer :+ Seq[KoreanPhrase]()))
255         case ((output, buffer), phrase) if isNonNounPhraseCandidate(phrase) =>
256           (output, addPhraseToBuffer(phrase, buffer))
257         case ((output, buffer), phrase) if buffer.length > 0 =>
258           (output ++ buffer, newBuffer)
259         case ((output, buffer), phrase) => (output, buffer)
260       }
261       if (buffer.length > 0) output ++ buffer else output
262     }

286   def extractPhrases(tokens: Seq[KoreanToken], filterSpam: Boolean): Seq[KoreanPhrase] = {
287     val collapsed = collapsePos(tokens)
288     val candidates = getCandidatePhraseChunks(collapsed, filterSpam)
289     val permutatedCandidates = permutateCadidates(candidates)
290 
291     permutatedCandidates.map {
292       phraseChunk: KoreanPhraseChunk => KoreanPhrase(trimPhraseChunk(phraseChunk).flatMap(_.tokens))
293     }
294   }

297   private def permutateCadidates(candidates: Seq[KoreanPhraseChunk]): Seq[KoreanPhraseChunk] = {
298     candidates.flatMap {
299       case phrases if phrases.length > MinPhrasesPerPhraseChunk =>
300         (0 to phrases.length - MinPhrasesPerPhraseChunk).map {
301           i => trimPhraseChunk(phrases.slice(i, phrases.length))
302         }
303       case phrases => Seq(phrases)
304     }.filter { phraseChunk: KoreanPhraseChunk => isProperPhraseChunk(phraseChunk)
305     }.distinct
306   }
