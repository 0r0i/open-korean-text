22   override def toString: String = s"$text($start,$end)"

25 object KoreanSentenceSplitter {
26   val re =
27     """(?x)[^.!?…\s]   # First char is non-punct, non-ws
28       [^.!?…]*         # Greedily consume up to punctuation.
29       (?:              # Group for unrolling the loop.
30         [.!?…]         # (special) inner punctuation ok if
31         (?!['\"]?\s|$) # not followed by ws or EOS.
32         [^.!?…]*       # Greedily consume up to punctuation.
33       )*               # Zero or more (special normal*)
34       [.!?…]?          # Optional ending punctuation.
35       ['\"]?           # Optional closing quote.
36       (?=\s|$)""".r
37 
38   def split(s: CharSequence) = {
39     re.findAllMatchIn(s)
40       .map(m => Sentence(m.group(0), m.start(0), m.end(0)))
41       .toSeq
42   }
43 }
