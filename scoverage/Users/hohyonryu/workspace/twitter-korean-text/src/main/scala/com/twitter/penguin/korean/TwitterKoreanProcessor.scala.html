50   /**
51    * Normalize Korean text. Uses KoreanNormalizer.normalize().
52    *
53    * @param text Input text
54    * @return Normalized Korean text
55    */

58   /**
59    * Wrapper for Korean stemmer
60    *
61    * @param text Input text
62    * @return A sequence of stemmed tokens
63    */
64   def stem(text: CharSequence): StemmedTextWithTokens = {
65     KoreanStemmer.stem(text)
66   }

68   /**
69    * Tokenize text into a sequence of token strings.
70    *
71    * @param text input text
72    * @param normalize option to enable the normalizer
73    * @param stem option to enable the stemmer
74    * @return A sequence of token strings.
75    */
76   def tokenizeToStrings(text: CharSequence, normalize: Boolean = true, stem: Boolean = true, keepSpace: Boolean = false): Seq[String] = {
77     tokenize(text, normalize, stem, keepSpace).map(_.text.toString)
78   }
79 
80   /**
81    * Tokenize text into a sequence of KoreanSegments, which includes start offset, the length,
82    * and the full information of each token.
83    *
84    * This is useful for Lucene integration. Normalization is not supported for this feature.
85    * For stemming support, use tokenizeWithIndexWithStemmer
86    *
87    * @param text Input text.
88    * @return A sequence of KoreanSegments.
89    */
90   def tokenizeWithIndex(text: CharSequence): Seq[KoreanSegment] = {
91     val tokens: Seq[KoreanToken] = tokenize(text, normalizization = false, stemming = false)
92     getKoreanSegments(text, tokens)
93   }
94 
95   /**
96    * Tokenize text into a KoreanSegmentWithText,
97    * which includes stemmed input text and a sequence stemmed tokens
98    *
99    * This is useful for Lucene integration. Normalization is not supported for this feature.
100    *
101    * @param text Input text.
102    * @return KoreanSegmentWithText
103    */
104   def tokenizeWithIndexWithStemmer(text: CharSequence): KoreanSegmentWithText = {
105     val stemmed: StemmedTextWithTokens = KoreanStemmer.stem(text)
106     KoreanSegmentWithText(stemmed.text, getKoreanSegments(stemmed.text, stemmed.tokens))
107   }
108 
109   /**
110    * Tokenize text into a sequence of KoreanTokens, which includes part-of-speech information and
111    * whether a token is an out-of-vocabulary term.
112    *
113    * @param text input text
114    * @param normalizization option to enable the normalizer
115    * @param stemming option to enable the stemmer
116    * @return A sequence of KoreanTokens.
117    */
118   def tokenize(text: CharSequence,
119                normalizization: Boolean = true,
120                stemming: Boolean = true,
121                keepSpace: Boolean = false): Seq[KoreanToken] = {
122     val normalized = if (normalizization) KoreanNormalizer.normalize(text) else text
123     val tokenized = KoreanTokenizer.tokenize(normalized, keepSpace)
124     if (stemming) KoreanStemmer.stemPredicates(tokenized) else tokenized
125   }
126 
127   def extractPhrases(text: CharSequence): Seq[CharSequence] = {
128     KoreanPhraseExtractor.extractPhrases(text)
129   }
130 
131   /**
132    * Align text with tokens to get indices (useful for Lucene integration)
133    * @param text Input CharSequence
134    * @param tokens A sequence of Tokens
135    * @return  A sequence of KoreanSegments
136    */
137   private def getKoreanSegments(text: CharSequence, tokens: Seq[KoreanToken]): Seq[KoreanSegment] = {
138     val s = text.toString
139     // Match text with the tokenization results to get offset and length of each token
140     val (output, i) = tokens.foldLeft(List[KoreanSegment](), 0) {
141       case ((l: List[KoreanSegment], i: Int), token: KoreanToken) =>
142         val segStart = s.indexOf(token.text, i)
143         (KoreanSegment(segStart, token.text.length, token) :: l, segStart + token.text.length)
144     }
145     output.reverse
146   }
