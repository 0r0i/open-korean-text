29 /**
30  * TwitterKoreanTokenizer provides error and slang tolerant Korean tokenization.
31  */
32 object TwitterKoreanProcessor {
33   /**
34    * Normalize Korean text. Uses KoreanNormalizer.normalize().
35    *
36    * @param text Input text
37    * @return Normalized Korean text
38    */
39   def normalize(text: CharSequence): CharSequence = KoreanNormalizer.normalize(text)
40 
41 
42   /**
43    * Tokenize text into a sequence of KoreanTokens, which includes part-of-speech information and
44    * whether a token is an out-of-vocabulary term.
45    *
46    * @param text input text
47    * @return A sequence of KoreanTokens.
48    */
49   def tokenize(text: CharSequence): Seq[KoreanToken] = KoreanTokenizer.tokenize(text)
50 
51   /**
52    * Wrapper for Korean stemmer
53    *
54    * @param tokens Korean tokens
55    * @return A sequence of stemmed tokens
56    */
57   def stem(tokens: Seq[KoreanToken]): Seq[KoreanToken] = KoreanStemmer.stem(tokens)
58 
59   /**
60    * Tokenize text into a sequence of token strings. This excludes spaces.
61    *
62    * @param tokens Korean tokens
63    * @return A sequence of token strings.
64    */
65   def tokensToStrings(tokens: Seq[KoreanToken]): Seq[String] = {
66     tokens.filterNot(t => t.pos == KoreanPos.Space).map(_.text.toString)
67   }
68 
69   /**
70    * Split input text into sentences.
71    *
72    * @param text input text
73    * @return A sequence of sentences.
74    */
75   def splitSentences(text: CharSequence): Seq[Sentence] = {
76     KoreanSentenceSplitter.split(text)
77   }
78 
79   /**
80    * Extract noun-phrases from Korean text
81    *
82    * @param tokens Korean tokens
83    * @param filterSpam true if spam/slang terms to be filtered out (default: false)
84    * @param enableHashtags true if #hashtags to be included (default: true)
85    * @return A sequence of extracted phrases
86    */
87   def extractPhrases(tokens: Seq[KoreanToken],
88                      filterSpam: Boolean = false,
89                      enableHashtags: Boolean = true): Seq[KoreanPhrase] = {
90     KoreanPhraseExtractor.extractPhrases(tokens, filterSpam, enableHashtags)
91   }
92 
93 }
