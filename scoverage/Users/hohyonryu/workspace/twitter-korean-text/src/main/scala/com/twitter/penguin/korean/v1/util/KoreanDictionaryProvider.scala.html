35   private[this] def readStreamByLine(stream: InputStream, filename: String): Iterator[String] = {
36     require(stream != null, "Resource not loaded: " + filename)

43   private[this] def readWordFreqs(filename: String): collection.mutable.Map[CharSequence, Float] = {
44     var freqMap: collection.mutable.Map[CharSequence, Float] =
45       new java.util.HashMap[CharSequence, Float]
46 
47     readFileByLineFromResources(filename).foreach {
48       case line => if (line.contains("\t")) {
49         val data = line.split("\t")
50         freqMap += (data(0) -> data(1).slice(0, 6).toFloat)
51       }
52     }
53     freqMap
54   }

56   private[this] def readWordMap(filename: String): Map[String, String] = {
57     readFileByLineFromResources(filename).filter {
58       case line: String => line.contains(" ")
59     }.map {
60       case line =>
61         val data = line.split(" ")
62         (data(0), data(1))
63     }.toMap
64   }

80   protected[korean] def readWords(filenames: String*): CharArraySet = {
