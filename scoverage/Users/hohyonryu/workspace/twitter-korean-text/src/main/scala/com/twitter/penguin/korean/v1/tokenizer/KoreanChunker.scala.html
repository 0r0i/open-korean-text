28 /**
29  * Split input text into Korean Chunks (어절)
30  */
31 object KoreanChunker {
32   val POS_PATTERNS = Map(
33     Korean -> """([가-힣]+)""".r.pattern,
34     Alpha -> """(\p{Alpha}+)""".r.pattern,
35     Number -> """(\p{Digit}+)""".r.pattern,
36     KoreanParticle -> """([ㄱ-ㅣ]+)""".r.pattern,
37     Punctuation -> """(\p{Punct}+)""".r.pattern,
38     URL -> Regex.VALID_URL,
39     Email -> """([\p{Alnum}\.\-_]+@[\p{Alnum}\.]+)""".r.pattern,
40     Hashtag -> Regex.VALID_HASHTAG,
41     ScreenName -> Regex.VALID_MENTION_OR_LIST,
42     CashTag -> Regex.VALID_CASHTAG
43   )
44 
45   val CHUNKING_ORDER = Seq(URL, Email, ScreenName, Hashtag, CashTag, Korean, KoreanParticle, Number, Alpha, Punctuation)
46   val SPACE_REGEX_STRING = "\\s"
47 
48   protected[korean] def getChunks(input: String): Seq[String] = {
49     chunk(input).map(_.text)
50   }
51 
52 
53   private[this] case class ChunkMatch(start: Int, end: Int, text: String, pos: KoreanPos) {
54     def disjoint(that: ChunkMatch): Boolean = {
55       (that.start < this.start && that.end <= this.start) ||
56           (that.start >= this.end && that.end > this.end)
57     }
58   }
59 
60   /**
61    * Recursively call m.find() to find all the matches.
62    * Use tail-recursion optimization to avoid stack overflow.
63    *
64    * @param m input Matcher
65    * @param pos KoreanPos to attach
66    * @param matches ouput list of ChunkMatch
67    * @return list of ChunkMatches
68    */
69   @scala.annotation.tailrec
70   private[this] def findAllPatterns(m: Matcher, pos: KoreanPos, matches: List[ChunkMatch] = List()): List[ChunkMatch] = {
71     if (m.find()) {
72       findAllPatterns(m, pos, ChunkMatch(m.start, m.end, m.group(), pos) :: matches)
73     } else {
74       matches
75     }
76   }
77 
78   private[this] def splitChunks(text: String): List[ChunkMatch] = {
79     val chunks = text.split(SPACE_REGEX_STRING).flatMap(s =>
80       CHUNKING_ORDER.foldLeft(List[ChunkMatch]()) {
81         (l, pos) =>
82           val m = POS_PATTERNS(pos).matcher(s)
83 
84           findAllPatterns(m, pos).filter(cm => l.forall(cm.disjoint)) ::: l
85       }
86     ).sortBy(cm => cm.start)
87 
88     fillInUnmatched(text, chunks, Foreign)
89   }
90 
91   /**
92    * Fill in unmatched segments with given pos
93    *
94    * @param text input text
95    * @param chunks matched chunks
96    * @param pos KoreanPos to attach to the unmatched chunk
97    * @return list of ChunkMatches
98    */
99   private[this] def fillInUnmatched(text: String,
100                                     chunks: Array[ChunkMatch],
101                                     pos: KoreanPos.Value): List[ChunkMatch] = {
102 
103     // Add Foreign for unmatched parts
104     val (chunksWithForeign, prevEnd) = chunks.foldLeft((List[ChunkMatch](), 0)) {
105       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start == prevEnd =>
106         (cm :: l, cm.end)
107       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start > prevEnd =>
108         (cm :: ChunkMatch(prevEnd, cm.start, text.slice(prevEnd, cm.start), pos) :: l, cm.end)
109       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) =>
110         throw new IllegalStateException("Non-disjoint chunk matches found.")
111     }
112 
113     val output = if (prevEnd < text.length) {
114       ChunkMatch(prevEnd, text.length, text.slice(prevEnd, text.length), pos) :: chunksWithForeign
115     } else {
116       chunksWithForeign
117     }
118 
119     output.reverse
120   }
121 
122 
123   /**
124    * Get chunks by given pos.
125    *
126    * @param input input string
127    * @param pos one of supported KoreanPos's: URL, Email, ScreenName, Hashtag,
128    *            CashTag, Korean, KoreanParticle, Number, Alpha, Punctuation
129    * @return sequence of Korean chunk strings
130    */
131   def getChunksByPos(input: String, pos: KoreanPos): Seq[String] = {
132     chunk(input).filter(_.pos == pos).map(_.text)
133   }
134 
135   /**
136    * Split input text into a sequnce of KoreanToken. A candidate for Korean parser
137    * gets tagged with KoreanPos.Korean.
138    *
139    * @param input input string
140    * @return sequence of KoreanTokens
141    */
142   def chunk(input: CharSequence): Seq[KoreanToken] = {
143     input.toString
144         .split(SPACE_REGEX_STRING)
145         .flatMap(splitChunks)
146         .map(m => KoreanToken(m.text, m.pos))
147   }
148 }
