28 /**
29  * Create Korean predicate expansion goldenset from adjective and verb resources.
30  * The first argument is the test resource directory.
31  */
32 object CreateConjugationGoldenset {
33 
34   case class ConjugationExample(word: String, conjugations: Seq[String])
35 
36   def main(args: Array[String]) {
37     System.err.println("Reading the verbs and adjectives..")
38 
39     def updateConjugateExamples(file: String, isAdj: Boolean, outputFileName: String) {
40       System.err.println("Writing the expansion goldenset in " + outputFileName)
41 
42       val outputPath = "src/test/resources/com/twitter/penguin/korean/v1/util/" + outputFileName
43       val out = new FileOutputStream(outputPath)
44 
45       val words = readWordsAsSeq(file)
46       val goldenset = words.map(word =>
47         ConjugationExample(word,
48           conjugatePredicates(Set(word), isAdj)
49               .map { case word: Array[Char] => new String(word)}
50               .toSeq.sorted)
51       )
52 
53       goldenset.foreach {
54         c => out.write(
55           "%s\t%s\n".format(c.word, c.conjugations.mkString(", ")).getBytes
56         )
57       }
58 
59       out.close()
60     }
61 
62     updateConjugateExamples("adjective/adjective.txt", isAdj = true, "adj_conjugate.txt")
63     updateConjugateExamples("verb/verb.txt", isAdj = false, "verb_conjugate.txt")
64   }
65 }
