54   /**
55    * A candidate parse for a chunk.
56    *
57    * @param posNodes Sequence of KoreanTokens.
58    * @param words Number of words in this candidate parse.
59    */
60   case class ParsedChunk(posNodes: Seq[KoreanToken], words: Int) {
61     def ++(that: ParsedChunk) = {
62       ParsedChunk(this.posNodes ++ that.posNodes, this.words + that.words)
63     }
64 
65     lazy val score = countTokens * WEIGHT_TOKENS +
66         countUnknowns * WEIGHT_UNKNOWNS +
67         words * WEIGHT_WORDS +
68         getUnknownCoverage * WEIGHT_UNKNOWN_COVERAGE +
69         getFreqScore * WEIGHT_FREQ +
70         countPos(Unknown) * WEIGHT_POS_UNKNOWNS +
71         isExactMatch * WEIGHT_EXACT_MATCH +
72         isAllNouns * WEIGHT_ALL_NOUN +
73         isPreferredPattern * WEIGHT_PREFFERED_PATTERN
74 
75     lazy val countUnknowns = this.posNodes.count { p: KoreanToken => p.unknown}
76     lazy val countTokens = this.posNodes.size
77 
78     lazy val isExactMatch = if (this.posNodes.size == 1) 0 else 1
79     lazy val isAllNouns = if (this.posNodes.exists(_.pos != Noun)) 1 else 0
80     lazy val isPreferredPattern = if (posNodes.size == 2 && posNodes.map(_.pos) == PREFERRED_PATTERN) 0 else 1
81 
82     lazy val posTieBreaker = this.posNodes.map(_.pos.id).sum
83 
84     lazy val getUnknownCoverage = this.posNodes.foldLeft(0) {
85       case (sum, p: KoreanToken) => if (p.unknown) sum + p.text.length else sum
86     }
87 
88     lazy val getFreqScore = this.posNodes.foldLeft(0f) {
89       case (output: Float, p: KoreanToken) if p.pos == Noun => output + (1f - koreanEntityFreq.getOrElse(p.text, 0f))
90       case (output: Float, p: KoreanToken) => output + 1.0f
91     } / this.posNodes.size
92 
93     def countPos(pos: KoreanPos) = this.posNodes.count { p: KoreanToken => p.pos == pos}
94   }

109   /**
110    * 0 for optional, 1 for required
111    * * for optional repeatable, + for required repeatable
112    *
113    * Substantive: 체언 (초거대기업의)
114    * Predicate: 용언 (하였었습니다, 개예뻤었다)
115    * Modifier: 수식언 (모르는 할수도있는 보이기도하는 예뻐 예쁜 완전 레알 초인간적인 잘 잘한)

117    * Functional: 관계언 (조사)
118    *
119    * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
120    * V Verb: 동사 (하, 먹, 자, 차)
121    * J Adjective: 형용사 (예쁘다, 크다, 작다)
122    * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
123    * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
124    * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
125    *
126    * C Conjunction: 접속사
127    *
128    * j SubstantiveJosa: 조사 (의, 에, 에서)
129    * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
130    * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
131    * r PreEomi: 선어말어미 (었)
132    *
133    * p NounPrefix: 접두사 ('초'대박)
134    * v VerbPrefix: 동사 접두어 ('쳐'먹어)

150   protected[korean] def buildTrie(s: String): List[KoreanPosTrie] = {
151     def isFinal(rest: String): Boolean = {
152       val isNextOptional = rest.foldLeft(true) {
153         case (output: Boolean, c: Char) if c == '+' || c == '1' => false
154         case (output: Boolean, c: Char) => output
155       }
156       rest.length == 0 || isNextOptional
157     }
158 
159     if (s.length < 2) {
160       return List()
161     }
162 
163     val pos = shortCut(s.charAt(0))
164     val rule = s.charAt(1)
165     val rest = if (s.length > 1) {
166       s.slice(2, s.length)
167     } else {
168       ""
169     }
170 
171     val end: Boolean = isFinal(rest)
172 
173     rule match {
174       case '+' =>
175         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest), end))
176       case '*' =>
177         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest), end)) ++ buildTrie(rest)
178       case '1' =>
179         List(KoreanPosTrie(pos, buildTrie(rest), end))
180       case '0' =>
181         List(KoreanPosTrie(pos, buildTrie(rest), end)) ++ buildTrie(rest)
182     }
183   }

196   /**
197    * Recursively find the best parse.
198    *
199    * @param chunk Input chunk. The input has to be entirely. Check for input validity is skipped
200    *              for performance optimization. This method is private and is called only by tokenize.
201    * @return The best possible parse.
202    */
203   private[this] def parseKoreanChunk(chunk: String): Seq[KoreanToken] = {
204 
205     // Buffer for solutions
206     val solutions: mutable.Map[Int, List[CandidateParse]] = new java.util.HashMap[Int, List[CandidateParse]]
207 
208     // Initial state
209     solutions += 0 -> List(
210       CandidateParse(
211         ParsedChunk(Seq[KoreanToken](), 1),
212         koreanPosTrie, ending = false
213       )
214     )
215 
216     // Find N best parses per state
217     for (
218       end <- 1 to chunk.length;
219       start <- end - 1 to(0, -1)
220     ) {
221       val word = chunk.slice(start, end)
222 
223       val curSolutions = solutions(start)
224 
225       val candidates = curSolutions.flatMap {
226         solution =>
227           val possiblePoses: Seq[PossibleTrie] = if (solution.ending) {
228             solution.curTrie.map(t => PossibleTrie(t, 0)) ++ koreanPosTrie.map(t => PossibleTrie(t, 1))
229           } else {
230             solution.curTrie.map(t => PossibleTrie(t, 0))
231           }
232 
233           possiblePoses.view.filter { t =>
234             t.curTrie.curPos == Noun || koreanDictionary(t.curTrie.curPos).contains(word.toCharArray)
235           }.map {
236             case t: PossibleTrie =>
237               val candidateToAdd =
238                 if (t.curTrie.curPos == Noun && !koreanDictionary(Noun).contains(word.toCharArray)) {
239                   val unknown = !isName(word) && !isKoreanNumber(word)
240                   ParsedChunk(Seq(KoreanToken(word, Noun, unknown)), t.words)
241                 } else {
242                   ParsedChunk(Seq(KoreanToken(word, t.curTrie.curPos)), t.words)
243                 }
244 
245               val nextTrie = t.curTrie.nextTrie.map {
246                 case nt: KoreanPosTrie if nt == selfNode => t.curTrie
247                 case nt: KoreanPosTrie => nt
248               }
249 
250               CandidateParse(solution.parse ++ candidateToAdd, nextTrie, t.curTrie.ending)
251           }
252       }
253 
254       val currentSolutions = if (solutions.contains(end)) solutions(end) else List()
255 
256       solutions += end -> (currentSolutions ++ candidates).sortBy {
257         c => (c.parse.score, c.parse.posTieBreaker)
258       }.take(TOP_N_PER_STATE)
259     }
260 
261     // Return the best parse of the final state
262     solutions(chunk.length).minBy(c => c.parse.score).parse.posNodes
263   }

265   /**
266    * Parse Korean text into a sequence of KoreanTokens
267    *
268    * @param text Input Korean chunk
269    * @return sequence of KoreanTokens
270    */
271   def tokenize(text: CharSequence): Seq[KoreanToken] = {
272     chunk(text).flatMap {
273       case token: KoreanToken if token.pos == Korean =>
274         // Get the best parse of each chunk
275         val parsed = parseKoreanChunk(token.text)
276 
277         // Collapse sequence of one-char nouns into one unknown noun: (가Noun 회Noun -> 가회Noun*)
278         collapseNouns(parsed)
279       case token: KoreanToken => Seq(token)
280     }
