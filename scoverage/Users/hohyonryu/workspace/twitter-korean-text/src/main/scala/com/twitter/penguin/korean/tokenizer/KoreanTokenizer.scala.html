30 /**
31  * Provides Korean tokenization.
32  *
33  * Chunk: 어절 - 공백으로 구분되어 있는 단위 (사랑하는사람을)
34  * Word: 단어 - 하나의 문장 구성 요소 (사랑하는, 사람을)
35  * Token: 토큰 - 형태소와 비슷한 단위이지만 문법적으로 정확하지는 않음 (사랑, 하는, 사람, 을)
36  *
37  * Whenever there is an updates in the behavior of KoreanParser,
38  * the initial cache has to be updated by running tools.CreateInitialCache.
39  */

58   /**
59    * A candidate parse for a chunk.
60    *
61    * @param posNodes Sequence of KoreanTokens.
62    * @param words Number of words in this candidate parse.
63    */
64   case class ParsedChunk(posNodes: Seq[KoreanToken], words: Int) {
65     def ++(that: ParsedChunk) = {
66       ParsedChunk(this.posNodes ++ that.posNodes, this.words + that.words)
67     }
68 
69     lazy val score = countTokens * WEIGHT_TOKENS +
70       countUnknowns * WEIGHT_UNKNOWNS +
71       words * WEIGHT_WORDS +
72       getUnknownCoverage * WEIGHT_UNKNOWN_COVERAGE +
73       getFreqScore * WEIGHT_FREQ +
74       countPos(Unknown) * WEIGHT_POS_UNKNOWNS +
75       isExactMatch * WEIGHT_EXACT_MATCH +
76       isAllNouns * WEIGHT_ALL_NOUN +
77       isPreferredPattern * WEIGHT_PREFFERED_PATTERN +
78       countPos(Determiner) * WEIGHT_DETERMINER
79 
80     lazy val countUnknowns = this.posNodes.count { p: KoreanToken => p.unknown }
81     lazy val countTokens = this.posNodes.size
82 
83     lazy val isExactMatch = if (this.posNodes.size == 1) 0 else 1
84     lazy val isAllNouns = if (this.posNodes.exists(t => t.pos != Noun && t.pos != ProperNoun)) 1 else 0
85     lazy val isPreferredPattern = if (
86       posNodes.size == 2 &&
87         (posNodes.map(_.pos) == PREFERRED_PATTERN_NOUN || posNodes.map(_.pos) == PREFERRED_PATTERN_PROPER_NOUN)
88     ) 0 else 1
89 
90     lazy val posTieBreaker = this.posNodes.map(_.pos.id).sum
91 
92     lazy val getUnknownCoverage = this.posNodes.foldLeft(0) {
93       case (sum, p: KoreanToken) => if (p.unknown) sum + p.text.length else sum
94     }
95 
96     lazy val getFreqScore = this.posNodes.foldLeft(0f) {
97       case (output: Float, p: KoreanToken) if p.pos == Noun || p.pos == ProperNoun =>
98         output + (1f - koreanEntityFreq.getOrElse(p.text, 0f))
99       case (output: Float, p: KoreanToken) => output + 1.0f
100     } / this.posNodes.size
101 
102     def countPos(pos: KoreanPos) = this.posNodes.count { p: KoreanToken => p.pos == pos }
103   }

116   /**
117    * 0 for optional, 1 for required
118    * * for optional repeatable, + for required repeatable
119    *
120    * Substantive: 체언 (초거대기업의)
121    * Predicate: 용언 (하였었습니다, 개예뻤었다)
122    * Modifier: 수식언 (모르는 할수도있는 보이기도하는 예뻐 예쁜 완전 레알 초인간적인 잘 잘한)
123    * Standalone: 독립언
124    * Functional: 관계언 (조사)
125    *
126    * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
127    * V Verb: 동사 (하, 먹, 자, 차)
128    * J Adjective: 형용사 (예쁘다, 크다, 작다)
129    * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
130    * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
131    * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
132    *
133    * C Conjunction: 접속사
134    *
135    * j SubstantiveJosa: 조사 (의, 에, 에서)
136    * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
137    * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
138    * r PreEomi: 선어말어미 (었)
139    *
140    * p NounPrefix: 접두사 ('초'대박)
141    * v VerbPrefix: 동사 접두어 ('쳐'먹어)
142    * s Suffix: 접미사 (~적)
143    */
144   private val SequenceDefinition = Map(
145     // Substantive
146     "D0p*N1s0j0" -> Noun,
147     // Predicate 초기뻐하다, 와주세요, 초기뻤었고, 추첨하다, 구경하기힘들다, 기뻐하는, 기쁜, 추첨해서, 좋아하다, 걸려있을
148     "v*V1r*e0" -> Verb,
149     "v*J1r*e0" -> Adjective,
150     // Modifier 부사
151     "A1" -> Adverb,
152     // Standalone
153     "C1" -> Conjunction,
154     "E+" -> Exclamation,
155     "j1" -> Josa
156   )

166   /**
167    * Find the best parse using dynamic programming.
168    *
169    * @param chunk Input chunk. The input has to be entirely. Check for input validity is skipped
170    *              for performance optimization. This method is private and is called only by tokenize.
171    * @return The best possible parse.
172    */
173   private[this] def parseKoreanChunk(chunk: KoreanToken): Seq[KoreanToken] = {
174 
175     // Direct match
176     koreanDictionary.foreach {
177       case (pos, dict) =>
178         if (dict.contains(chunk.text)) {
179           return Seq(KoreanToken(chunk.text, pos, chunk.offset, chunk.length))
180         }
181     }
182 
183     // Buffer for solutions
184     val solutions: mutable.Map[Int, List[CandidateParse]] = new java.util.HashMap[Int, List[CandidateParse]]
185 
186     // Initial state
187     solutions += 0 -> List(
188       CandidateParse(
189         ParsedChunk(Seq[KoreanToken](), 1),
190         koreanPosTrie, ending = None
191       )
192     )
193 
194     // Find N best parses per state
195     for (
196       end <- 1 to chunk.length;
197       start <- end - 1 to(Seq(end - MAX_TRACE_BACK, 0).max, -1)
198     ) {
199       val word = chunk.text.slice(start, end)
200 
201       val curSolutions = solutions(start)
202 
203       val candidates = curSolutions.flatMap {
204         solution =>
205           val possiblePoses: Seq[PossibleTrie] = if (solution.ending.isDefined) {
206             solution.curTrie.map(t => PossibleTrie(t, 0)) ++ koreanPosTrie.map(t => PossibleTrie(t, 1))
207           } else {
208             solution.curTrie.map(t => PossibleTrie(t, 0))
209           }
210 
211           possiblePoses.view.filter { t =>
212             t.curTrie.curPos == Noun || koreanDictionary(t.curTrie.curPos).contains(word.toCharArray)
213           }.map { case t: PossibleTrie =>
214             val candidateToAdd =
215               if (t.curTrie.curPos == Noun && !koreanDictionary(Noun).contains(word.toCharArray)) {
216                 val isWordName: Boolean = isName(word)
217                 val isWordKoreanNameVariation: Boolean = isKoreanNameVariation(word)
218 
219                 val unknown = !isWordName && !isKoreanNumber(word) && !isWordKoreanNameVariation
220                 val pos = if (unknown || isWordName || isWordKoreanNameVariation) ProperNoun else Noun
221                 ParsedChunk(Seq(KoreanToken(word, pos, chunk.offset + start, word.length, unknown)), t.words)
222               } else {
223                 val pos = if (t.curTrie.curPos == Noun && properNouns.contains(word.toCharArray)) ProperNoun else t.curTrie.curPos
224                 ParsedChunk(Seq(KoreanToken(word, pos, chunk.offset + start, word.length)), t.words)
225               }
226 
227             val nextTrie = t.curTrie.nextTrie.map {
228               case nt: KoreanPosTrie if nt == selfNode => t.curTrie
229               case nt: KoreanPosTrie => nt
230             }
231 
232             CandidateParse(solution.parse ++ candidateToAdd, nextTrie, t.curTrie.ending)
233           }
234       }
235 
236       val currentSolutions = if (solutions.contains(end)) solutions(end) else List()
237 
238       solutions += end -> (currentSolutions ++ candidates).sortBy {
239         c => (c.parse.score, c.parse.posTieBreaker)
240       }.take(TOP_N_PER_STATE)
241     }
242 
243     // Return the best parse of the final state
244     solutions(chunk.length).minBy(c => c.parse.score).parse.posNodes
245   }

247   /**
248    * Parse Korean text into a sequence of KoreanTokens
249    *
250    * @param text Input Korean chunk
251    * @return sequence of KoreanTokens
252    */
253   def tokenize(text: CharSequence): Seq[KoreanToken] = {
254     chunk(text).flatMap {
255       case token: KoreanToken if token.pos == Korean =>
256         // Get the best parse of each chunk
257         val parsed = parseKoreanChunk(token)
258 
259         // Collapse sequence of one-char nouns into one unknown noun: (가Noun 회Noun -> 가회Noun*)
260         collapseNouns(parsed)
261       case token: KoreanToken => Seq(token)
262     }
263   }
