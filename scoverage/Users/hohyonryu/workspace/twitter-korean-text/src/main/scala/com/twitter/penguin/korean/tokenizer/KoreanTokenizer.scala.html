30 /**
31  * Provides Korean tokenization.
32  *
33  * Chunk: 어절 - 공백으로 구분되어 있는 단위 (사랑하는사람을)
34  * Word: 단어 - 하나의 문장 구성 요소 (사랑하는, 사람을)
35  * Token: 토큰 - 형태소와 비슷한 단위이지만 문법적으로 정확하지는 않음 (사랑, 하는, 사람, 을)
36  *
37  * Whenever there is an updates in the behavior of KoreanParser,
38  * the initial cache has to be updated by running tools.CreateInitialCache.
39  */

57   /**
58    * A candidate parse for a chunk.
59    *
60    * @param posNodes Sequence of KoreanTokens.
61    * @param words Number of words in this candidate parse.
62    */
63   case class ParsedChunk(posNodes: Seq[KoreanToken], words: Int) {
64     def ++(that: ParsedChunk) = {
65       ParsedChunk(this.posNodes ++ that.posNodes, this.words + that.words)
66     }
67 
68     lazy val score = countTokens * WEIGHT_TOKENS +
69         countUnknowns * WEIGHT_UNKNOWNS +
70         words * WEIGHT_WORDS +
71         getUnknownCoverage * WEIGHT_UNKNOWN_COVERAGE +
72         getFreqScore * WEIGHT_FREQ +
73         countPos(Unknown) * WEIGHT_POS_UNKNOWNS +
74         isExactMatch * WEIGHT_EXACT_MATCH +
75         isAllNouns * WEIGHT_ALL_NOUN +
76         isPreferredPattern * WEIGHT_PREFFERED_PATTERN +
77         countPos(Determiner) * WEIGHT_DETERMINER
78 
79     lazy val countUnknowns = this.posNodes.count { p: KoreanToken => p.unknown}
80     lazy val countTokens = this.posNodes.size
81 
82     lazy val isExactMatch = if (this.posNodes.size == 1) 0 else 1
83     lazy val isAllNouns = if (this.posNodes.exists(_.pos != Noun)) 1 else 0
84     lazy val isPreferredPattern = if (posNodes.size == 2 && posNodes.map(_.pos) == PREFERRED_PATTERN) 0 else 1
85 
86     lazy val posTieBreaker = this.posNodes.map(_.pos.id).sum
87 
88     lazy val getUnknownCoverage = this.posNodes.foldLeft(0) {
89       case (sum, p: KoreanToken) => if (p.unknown) sum + p.text.length else sum
90     }
91 
92     lazy val getFreqScore = this.posNodes.foldLeft(0f) {
93       case (output: Float, p: KoreanToken) if p.pos == Noun => output + (1f - koreanEntityFreq.getOrElse(p.text, 0f))
94       case (output: Float, p: KoreanToken) => output + 1.0f
95     } / this.posNodes.size
96 
97     def countPos(pos: KoreanPos) = this.posNodes.count { p: KoreanToken => p.pos == pos}
98   }

111   /**
112    * 0 for optional, 1 for required
113    * * for optional repeatable, + for required repeatable
114    *
115    * Substantive: 체언 (초거대기업의)
116    * Predicate: 용언 (하였었습니다, 개예뻤었다)
117    * Modifier: 수식언 (모르는 할수도있는 보이기도하는 예뻐 예쁜 완전 레알 초인간적인 잘 잘한)
118    * Standalone: 독립언
119    * Functional: 관계언 (조사)
120    *
121    * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
122    * V Verb: 동사 (하, 먹, 자, 차)
123    * J Adjective: 형용사 (예쁘다, 크다, 작다)
124    * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
125    * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
126    * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
127    *
128    * C Conjunction: 접속사
129    *
130    * j SubstantiveJosa: 조사 (의, 에, 에서)
131    * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
132    * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
133    * r PreEomi: 선어말어미 (었)
134    *
135    * p NounPrefix: 접두사 ('초'대박)
136    * v VerbPrefix: 동사 접두어 ('쳐'먹어)
137    * s Suffix: 접미사 (~적)
138    */
139   private val SequenceDefinition = Map(
140     // Substantive
141     "D0p*N1s0j0" -> Noun,
142     // Predicate 초기뻐하다, 와주세요, 초기뻤었고, 추첨하다, 구경하기힘들다, 기뻐하는, 기쁜, 추첨해서, 좋아하다, 걸려있을
143     "v*V1r*e0" -> Verb,
144     "v*J1r*e0" -> Adjective,
145     // Modifier 부사
146     "A1" -> Adverb,
147     // Standalone
148     "C1" -> Conjunction,
149     "E+" -> Exclamation,
150     "j1" -> Josa
151   )

161   /**
162    * Find the best parse using dynamic programming.
163    *
164    * @param chunk Input chunk. The input has to be entirely. Check for input validity is skipped
165    *              for performance optimization. This method is private and is called only by tokenize.
166    * @return The best possible parse.
167    */
168   private[this] def parseKoreanChunk(chunk: String): Seq[KoreanToken] = {
169 
170     // Direct match
171     koreanDictionary.foreach {
172       case (pos, dict) =>
173         if (dict.contains(chunk)) {
174           return Seq(KoreanToken(chunk, pos))
175         }
176     }
177 
178     // Buffer for solutions
179     val solutions: mutable.Map[Int, List[CandidateParse]] = new java.util.HashMap[Int, List[CandidateParse]]
180 
181     // Initial state
182     solutions += 0 -> List(
183       CandidateParse(
184         ParsedChunk(Seq[KoreanToken](), 1),
185         koreanPosTrie, ending = None
186       )
187     )
188 
189     // Find N best parses per state
190     for (
191       end <- 1 to chunk.length;
192       start <- end - 1 to(Seq(end - MAX_TRACE_BACK, 0).max, -1)
193     ) {
194       val word = chunk.slice(start, end)
195 
196       val curSolutions = solutions(start)
197 
198       val candidates = curSolutions.flatMap {
199         solution =>
200           val possiblePoses: Seq[PossibleTrie] = if (solution.ending.isDefined) {
201             solution.curTrie.map(t => PossibleTrie(t, 0)) ++ koreanPosTrie.map(t => PossibleTrie(t, 1))
202           } else {
203             solution.curTrie.map(t => PossibleTrie(t, 0))
204           }
205 
206           possiblePoses.view.filter { t =>
207             t.curTrie.curPos == Noun || koreanDictionary(t.curTrie.curPos).contains(word.toCharArray)
208           }.map { case t: PossibleTrie =>
209             val candidateToAdd =
210               if (t.curTrie.curPos == Noun && !koreanDictionary(Noun).contains(word.toCharArray)) {
211                 val unknown = !isName(word) && !isKoreanNumber(word) && !isKoreanNameVariation(word)
212                 ParsedChunk(Seq(KoreanToken(word, Noun, unknown)), t.words)
213               } else {
214                 ParsedChunk(Seq(KoreanToken(word, t.curTrie.curPos)), t.words)
215               }
216 
217             val nextTrie = t.curTrie.nextTrie.map {
218               case nt: KoreanPosTrie if nt == selfNode => t.curTrie
219               case nt: KoreanPosTrie => nt
220             }
221 
222             CandidateParse(solution.parse ++ candidateToAdd, nextTrie, t.curTrie.ending)
223           }
224       }
225 
226       val currentSolutions = if (solutions.contains(end)) solutions(end) else List()
227 
228       solutions += end -> (currentSolutions ++ candidates).sortBy {
229         c => (c.parse.score, c.parse.posTieBreaker)
230       }.take(TOP_N_PER_STATE)
231     }
232 
233     // Return the best parse of the final state
234     solutions(chunk.length).minBy(c => c.parse.score).parse.posNodes
235   }

237   /**
238    * Parse Korean text into a sequence of KoreanTokens
239    *
240    * @param text Input Korean chunk
241    * @return sequence of KoreanTokens
242    */
243   def tokenize(text: CharSequence, keepSpace: Boolean = false): Seq[KoreanToken] = {
244     chunk(text, keepSpace).flatMap {
245       case token: KoreanToken if token.pos == Korean =>
246         // Get the best parse of each chunk
247         val parsed = parseKoreanChunk(token.text)
248 
249         // Collapse sequence of one-char nouns into one unknown noun: (가Noun 회Noun -> 가회Noun*)
250         collapseNouns(parsed)
251       case token: KoreanToken => Seq(token)
252     }
253   }
