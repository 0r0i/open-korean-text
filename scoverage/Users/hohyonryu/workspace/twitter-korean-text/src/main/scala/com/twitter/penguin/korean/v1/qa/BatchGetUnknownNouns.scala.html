30 /**
31  * Find unknown nouns from a file of tweets.
32  *
33  * usage: ./pants goal run src/scala/com/twitter/penguin/korean/qa:batch_get_unknown_nouns
34  * --jvm-run-args="~/korean_201407_100"
35  */
36 object BatchGetUnknownNouns {
37   val LOG = Logger.getLogger(getClass.getSimpleName)
38   val VERBOSE = true
39 
40   case class ChunkWithTweet(chunk: String, tweet: String)
41 
42   def main(args: Array[String]) {
43     if (args.length != 1) {
44       println("The first arg should be an input file path of Korean tweets.")
45       return
46     }
47     val chunksWithUnknowns = Source.fromFile(args(0)).getLines().foldLeft(List[ChunkWithTweet]()) {
48       case (l: List[ChunkWithTweet], line: String) if line.trim.length > 5 =>
49         chunk(line).flatMap {
50           case t: KoreanToken if t.pos == KoreanPos.Korean && tokenize(t.text).exists(_.unknown) =>
51             Some(ChunkWithTweet(t.text, line.trim))
52           case t: KoreanToken => None
53         }.toList ::: l
54       case (l: List[ChunkWithTweet], line: String) => l
55     }.toSet
56 
57     chunksWithUnknowns.toSeq.sortBy(_.chunk).foreach {
58       chunkWithTweet: ChunkWithTweet =>
59         println(chunkWithTweet.tweet)
60         println(TwitterKoreanProcessor
61             .tokenize(chunkWithTweet.tweet)
62             .mkString(" "))
63 
64         println(chunkWithTweet.chunk + ": " +
65             tokenize(chunkWithTweet.chunk).mkString(" "))
66         println()
67     }
68 
69   }
70 }
