30 /**
31  * Find unknown nouns from a file of tweets.
32  *
33  * Program arguments example = "~/korean_201407_100"
34  */
35 object BatchGetUnknownNouns {
36   private val LOG = Logger.getLogger(getClass.getSimpleName)
37   private val VERBOSE = true
38 
39   case class ChunkWithTweet(chunk: String, tweet: String)
40 
41   def main(args: Array[String]) {
42     if (args.length != 1) {
43       println("The first arg should be an input file path of Korean tweets.")
44       return
45     }
46     val chunksWithUnknowns = Source.fromFile(args(0)).getLines().foldLeft(List[ChunkWithTweet]()) {
47       case (l: List[ChunkWithTweet], line: String) if line.trim.length > 5 =>
48         chunk(line).flatMap {
49           case t: KoreanToken if t.pos == KoreanPos.Korean && tokenize(t.text).exists(_.unknown) =>
50             Some(ChunkWithTweet(t.text, line.trim))
51           case t: KoreanToken => None
52         }.toList ::: l
53       case (l: List[ChunkWithTweet], line: String) => l
54     }.toSet
55 
56     chunksWithUnknowns.toSeq.sortBy(_.chunk).foreach {
57       chunkWithTweet: ChunkWithTweet =>
58         println(chunkWithTweet.tweet)
59         println(OpenKoreanTextProcessor
60             .tokenize(chunkWithTweet.tweet)
61             .mkString(" "))
62 
63         println(chunkWithTweet.chunk + ": " +
64             tokenize(chunkWithTweet.chunk).mkString(" "))
65         println()
66     }
67 
68   }
69 }
