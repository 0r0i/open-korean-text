36   protected[processor] def isJosaAttachable(prevChar: Char, headChar: Char): Boolean = {
37     (hasCoda(prevChar) && !JOSA_HEAD_FOR_NO_CODA.contains(headChar)) ||
38         (!hasCoda(prevChar) && !JOSA_HEAD_FOR_CODA.contains(headChar))

41   protected[processor] def isName(chunk: CharSequence): Boolean = {
42     if (nameDictionary('full_name).contains(chunk) ||
43       nameDictionary('given_name).contains(chunk)) return true
44 
45     chunk.length match {
46       case 3 => {
47         (nameDictionary('family_name).contains(chunk.charAt(0).toString) &&
48         nameDictionary('given_name).contains(chunk.subSequence(1, 3).toString))
49       }
50       case 4 => {
51         (nameDictionary('family_name).contains(chunk.subSequence(0, 2).toString) &&
52           nameDictionary('given_name).contains(chunk.subSequence(2, 4).toString))
53       }
54       case _ => false
55     }
56   }

61   protected[processor] def isKoreanNumber(chunk: CharSequence): Boolean =
62     (0 to chunk.length() - 1).foldLeft(true) {
63       case (output, i) if i < chunk.length() - 1 => output && NUMBER_CHARS.contains(chunk.charAt(i).toInt)
64       case (output, i) => output && NUMBER_LAST_CHARS.contains(chunk.charAt(i).toInt)
65     }

67   /**
68    * Check if this chunk is an 'ㅇ' omitted variation of a noun (우혀니 -> 우현, 우현이, 빠순이 -> 빠순, 빠순이)
69    *
70    * @param chunk input chunk
71    * @return true if the chunk is an 'ㅇ' omitted variation
72    */
73   protected[processor] def isKoreanNameVariation(chunk: CharSequence): Boolean = {
74     val nounDict = koreanDictionary(Noun)
75 
76     val s = chunk.toString
77     if (isName(s)) return true
78     if (s.length < 3 || s.length > 5) return false
79 
80     val decomposed = s.map { c: Char => decomposeHangul(c)}
81     val lastChar = decomposed.last
82     if (!Hangul.CODA_MAP.contains(lastChar.onset)) return false
83     if (lastChar.onset == 'ㅇ' || lastChar.vowel != 'ㅣ' || lastChar.coda != ' ') return false
84     if (decomposed.init.last.coda != ' ') return false
85 
86     // Recover missing 'ㅇ' (우혀니 -> 우현, 우현이, 빠순이 -> 빠순, 빠순이)
87     val recovered = decomposed.zipWithIndex.map {
88       case (hc: HangulChar, i: Int) if i == s.length - 1 => '이'
89       case (hc: HangulChar, i: Int) if i == s.length - 2 =>
90         composeHangul(HangulChar(hc.onset, hc.vowel, decomposed.last.onset))
91       case (hc: HangulChar, i: Int) => composeHangul(hc)
92     }.mkString("")
93 
94     Seq(recovered, recovered.init).exists(isName)
95   }

97   /**
98    * Collapse all the one-char nouns into one unknown noun
99    *
100    * @param posNodes sequence of KoreanTokens
101    * @return sequence of collapsed KoreanTokens
102    */
103   protected[processor] def collapseNouns(posNodes: Seq[KoreanToken]): Seq[KoreanToken] = {
104     val (nodes, collapsing) = posNodes.foldLeft((List[KoreanToken](), false)) {
105       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken)
106         if p.pos == Noun && p.text.length == 1 && collapsing =>
107         val text = pl.head.text + p.text
108         val offset = pl.head.offset
109         (KoreanToken(text, Noun, offset, text.length, unknown = true) :: pl.tail, true)
110       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken)
111         if p.pos == Noun && p.text.length == 1 && !collapsing =>
112         (p :: pl, true)
113       case ((pl: List[KoreanToken], collapsing: Boolean), p: KoreanToken) =>
114         (p :: pl, false)
115     }
116     nodes.reverse.toSeq
117   }
