21 /**
22  * Korean Part-of-Speech
23  *
24  * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
25  * V Verb: 동사 (하, 먹, 자, 차)
26  * J Adjective: 형용사 (예쁘다, 크다, 작다)
27  * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
28  * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
29  * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
30  *
31  * C Conjunction: 접속사
32  *
33  * j SubstantiveJosa: 조사 (의, 에, 에서)
34  * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
35  * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
36  * r PreEomi: 선어말어미 (었)
37  *
38  * p NounPrefix: 접두사 ('초'대박)
39  * v VerbPrefix: 동사 접두어 ('쳐'먹어)
40  * s Suffix: 접미사 (~적)
41  *
42  * f Foreign: 한글이 아닌 문자들
43  *
44  * 지시사는 Derterminant로 대체하기로 함
45  * Derterminant is used for demonstratives.
46  *
47  * Korean: Korean chunk (candidate for parsing)
48  * Foreign: Mixture of non-Korean strings
49  * Number: 숫자
50  * Emotion: Korean Single Character Emotions (ㅋㅋㅋㅋ, ㅎㅎㅎㅎ, ㅠㅜㅠㅜ)
51  * Alpha: Alphabets 알파벳
52  * Punctuation: 문장부호
53  * Hashtag: Twitter Hashtag 해쉬태그 #Korean
54  * ScreenName: Twitter username (@nlpenguin)
55  *
56  * Unkown: Could not parse the string.
57  */
58 object KoreanPos extends Enumeration {
59   type KoreanPos = Value
60 
61   // Word leved POS
62   val Noun, Verb, Adjective,
63   Adverb, Determiner, Exclamation,
64   Josa, Eomi, PreEomi, Conjunction,
65   Modifier, VerbPrefix, Suffix, Unknown,
66 
67   // Chunk level POS
68   Korean, Foreign, Number, KoreanParticle, Alpha,
69   Punctuation, Hashtag, ScreenName,
70   Email, URL, CashTag,
71 
72   // Functional POS
73   Space, Others,
74 
75   ProperNoun = Value
76 
77   val OtherPoses = Set(
78     Korean, Foreign, Number, KoreanParticle, Alpha,
79     Punctuation, Hashtag, ScreenName,
80     Email, URL, CashTag)
81 
82   val shortCut = Map(
83     'N' -> Noun,
84     'V' -> Verb,
85     'J' -> Adjective,
86     'A' -> Adverb,
87     'D' -> Determiner,
88     'E' -> Exclamation,
89     'C' -> Conjunction,
90 
91     'j' -> Josa,
92     'e' -> Eomi,
93     'r' -> PreEomi,
94     'm' -> Modifier,
95     'v' -> VerbPrefix,
96     's' -> Suffix,
97 
98     'a' -> Alpha,
99     'n' -> Number,
100 
101     'o' -> Others
102   )
103 
104   case class KoreanPosTrie(curPos: KoreanPos, nextTrie: List[KoreanPosTrie], ending: Option[KoreanPos])
105 
106   val selfNode = KoreanPosTrie(null, null, ending = None)
107 
108   protected[processor] def buildTrie(s: String, ending_pos: KoreanPos): List[KoreanPosTrie] = {
109     def isFinal(rest: String): Boolean = {
110       val isNextOptional = rest.foldLeft(true) {
111         case (output: Boolean, c: Char) if c == '+' || c == '1' => false
112         case (output: Boolean, c: Char) => output
113       }
114       rest.length == 0 || isNextOptional
115     }
116 
117     if (s.length < 2) {
118       return List()
119     }
120 
121     val pos = shortCut(s.charAt(0))
122     val rule = s.charAt(1)
123     val rest = if (s.length > 1) {
124       s.slice(2, s.length)
125     } else {
126       ""
127     }
128 
129     val end: Option[KoreanPos] = if (isFinal(rest)) Some(ending_pos) else None
130 
131     rule match {
132       case '+' =>
133         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest, ending_pos), end))
134       case '*' =>
135         List(KoreanPosTrie(pos, selfNode :: buildTrie(rest, ending_pos), end)) ++ buildTrie(rest, ending_pos)
136       case '1' =>
137         List(KoreanPosTrie(pos, buildTrie(rest, ending_pos), end))
138       case '0' =>
139         List(KoreanPosTrie(pos, buildTrie(rest, ending_pos), end)) ++ buildTrie(rest, ending_pos)
140     }
141   }
142 
143   protected[processor] def getTrie(sequences: Map[String, KoreanPos]): List[KoreanPosTrie] =
144     sequences.foldLeft(List[KoreanPosTrie]()) {
145       case (results: List[KoreanPosTrie], (s: String, ending_pos: KoreanPos)) =>
146         buildTrie(s, ending_pos) ::: results
147     }
148 
149   val Predicates = Set(Verb, Adjective)
150 }
