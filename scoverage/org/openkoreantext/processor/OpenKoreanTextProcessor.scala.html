29 /**
30   * OpenKoreanTextProcessor provides error and slang tolerant Korean tokenization.
31   */
32 object OpenKoreanTextProcessor {
33   /**
34     * Normalize Korean text. Uses KoreanNormalizer.normalize().
35     *
36     * @param text Input text
37     * @return Normalized Korean text
38     */
39   def normalize(text: CharSequence): CharSequence = KoreanNormalizer.normalize(text)
40 
41 
42   /**
43     * Tokenize text into a sequence of KoreanTokens, which includes part-of-speech information and
44     * whether a token is an out-of-vocabulary term.
45     *
46     * @param text input text
47     * @return A sequence of KoreanTokens.
48     */
49   def tokenize(text: CharSequence): Seq[KoreanToken] = KoreanTokenizer.tokenize(text)
50 
51   /**
52     * Tokenize text (with a custom profile) into a sequence of KoreanTokens,
53     * which includes part-of-speech information and whether a token is an out-of-vocabulary term.
54     *
55     * @param text input text
56     * @param profile TokenizerProfile
57     * @return A sequence of KoreanTokens.
58     */
59   def tokenize(text: CharSequence,
60                profile: TokenizerProfile
61               ): Seq[KoreanToken] = {
62     KoreanTokenizer.tokenize(text, profile)
63   }
64 
65   /**
66     * Tokenize text (with a custom profile) into a sequence of KoreanTokens,
67     * which includes part-of-speech information and whether a token is an out-of-vocabulary term,
68     * and return top `n` candidates.
69     *
70     * @param text input text
71     * @param n number of top candidates
72     * @return A sequence of sequences of KoreanTokens.
73     */
74   def tokenizeTopN(text: CharSequence,
75                    n: Int
76                   ): Seq[Seq[Seq[KoreanToken]]] = {
77     KoreanTokenizer.tokenizeTopN(text, n)
78   }
79 
80   /**
81     * Tokenize text (with a custom profile) into a sequence of KoreanTokens,
82     * which includes part-of-speech information and whether a token is an out-of-vocabulary term,
83     * and return top `n` candidates.
84     *
85     * @param text input text
86     * @param n number of top candidates
87     * @param profile TokenizerProfile
88     * @return A sequence of sequences of KoreanTokens.
89     */
90   def tokenizeTopN(text: CharSequence,
91                    n: Int,
92                    profile: TokenizerProfile
93                   ): Seq[Seq[Seq[KoreanToken]]] = {
94     KoreanTokenizer.tokenizeTopN(text, n, profile)
95   }
96 
97   /**
98     * Add user-defined word list to the noun dictionary. Spaced words are not allowed.
99     *
100     * @param words Sequence of words to add.
101     */
102   def addNounsToDictionary(words: Seq[String]) {
103     KoreanDictionaryProvider.addWordsToDictionary(KoreanPos.Noun, words)
104   }
105 
106   /**
107     * Tokenize text into a sequence of token strings. This excludes spaces.
108     *
109     * @param tokens Korean tokens
110     * @return A sequence of token strings.
111     */
112   def tokensToStrings(tokens: Seq[KoreanToken]): Seq[String] = {
113     tokens.filterNot(t => t.pos == KoreanPos.Space).map(_.text.toString)
114   }
115 
116   /**
117     * Split input text into sentences.
118     *
119     * @param text input text
120     * @return A sequence of sentences.
121     */
122   def splitSentences(text: CharSequence): Seq[Sentence] = {
123     KoreanSentenceSplitter.split(text)
124   }
125 
126   /**
127     * Extract noun-phrases from Korean text
128     *
129     * @param tokens         Korean tokens
130     * @param filterSpam     true if spam/slang terms to be filtered out (default: false)
131     * @param enableHashtags true if #hashtags to be included (default: true)
132     * @return A sequence of extracted phrases
133     */
134   def extractPhrases(tokens: Seq[KoreanToken],
135                      filterSpam: Boolean = false,
136                      enableHashtags: Boolean = true): Seq[KoreanPhrase] = {
137     KoreanPhraseExtractor.extractPhrases(tokens, filterSpam, enableHashtags)
138   }
139 
140   /**
141     * Detokenize the input list of words.
142     *
143     * @param tokens List of words.
144     * @return Detokenized string.
145     */
146   def detokenize(tokens: Iterable[String]): String = {
147     KoreanDetokenizer.detokenize(tokens)
148   }
149 }
