30 /**
31   * Provides Korean tokenization.
32   *
33   * Chunk: 어절 - 공백으로 구분되어 있는 단위 (사랑하는사람을)
34   * Word: 단어 - 하나의 문장 구성 요소 (사랑하는, 사람을)
35   * Token: 토큰 - 형태소와 비슷한 단위이지만 문법적으로 정확하지는 않음 (사랑, 하는, 사람, 을)
36   *
37   * Whenever there is an updates in the behavior of KoreanParser,
38   * the initial cache has to be updated by running tools.CreateInitialCache.

43   /**
44     * 0 for optional, 1 for required
45     * * for optional repeatable, + for required repeatable
46     *
47     * Substantive: 체언 (초거대기업의)
48     * Predicate: 용언 (하였었습니다, 개예뻤었다)
49     * Modifier: 수식언 (모르는 할수도있는 보이기도하는 예뻐 예쁜 완전 레알 초인간적인 잘 잘한)
50     * Standalone: 독립언
51     * Functional: 관계언 (조사)
52     *
53     * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
54     * V Verb: 동사 (하, 먹, 자, 차)
55     * J Adjective: 형용사 (예쁘다, 크다, 작다)
56     * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
57     * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
58     * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
59     *
60     * C Conjunction: 접속사
61     *
62     * j SubstantiveJosa: 조사 (의, 에, 에서)
63     * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
64     * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
65     * r PreEomi: 선어말어미 (었)
66     *
67     * p NounPrefix: 접두사 ('초'대박)
68     * v VerbPrefix: 동사 접두어 ('쳐'먹어)
69     * s Suffix: 접미사 (~적)
70     */

98   /**
99     * Parse Korean text into a sequence of KoreanTokens with custom parameters
100     *
101     * @param text Input Korean chunk
102     * @return sequence of KoreanTokens
103     */
104   def tokenizeTopN(text: CharSequence,
105                    topN: Int = 1,
106                    profile: TokenizerProfile = TokenizerProfile.defaultProfile
107                   ): Seq[Seq[Seq[KoreanToken]]] = {
108     try
109       chunk(text).map {
110         case token: KoreanToken if token.pos == Korean =>
111           // Get the best parse of each chunk
112           val parsed = parseKoreanChunk(token, profile, topN)
113 
114           // Collapse sequence of one-char nouns into one unknown noun: (가Noun 회Noun -> 가회Noun*)
115           parsed.map(collapseNouns(_))
116         case token: KoreanToken => Seq(Seq(token))
117       }
118     catch {
119       case e: Exception =>
120         System.err.println(s"Error tokenizing a chunk: $text")
121         throw e
122     }
123   }

125   /**
126     * Find the best parse using dynamic programming.
127     *
128     * @param chunk Input chunk. The input has to be entirely. Check for input validity is skipped
129     *              for performance optimization. This method is private and is called only by tokenize.
130     * @return The best possible parse.
131     */
132   private[this] def parseKoreanChunk(chunk: KoreanToken,
133                                      profile: TokenizerProfile = TokenizerProfile.defaultProfile,
134                                      topN: Int = 1
135                                     ): Seq[Seq[KoreanToken]] = {
136     findTopCandidates(chunk, profile).take(topN)
137   }

139   private def findTopCandidates(chunk: KoreanToken, profile: TokenizerProfile) = {
140     val directMatch: Seq[Seq[KoreanToken]] = findDirectMatch(chunk)
141 
142     // Buffer for solutions
143     val solutions: mutable.Map[Int, List[CandidateParse]] = new java.util.HashMap[Int, List[CandidateParse]]
144 
145     // Initial state
146     solutions += 0 -> List(
147       CandidateParse(
148         ParsedChunk(Seq[KoreanToken](), 1, profile),
149         koreanPosTrie, ending = None
150       )
151     )
152 
153     // Find N best parses per state
154     for (
155       end <- 1 to chunk.length;
156       start <- end - 1 to(Seq(end - MAX_TRACE_BACK, 0).max, -1)
157     ) {
158       val word = chunk.text.slice(start, end)
159 
160       val curSolutions = solutions(start)
161 
162       val candidates = curSolutions.flatMap {
163         solution =>
164           val possiblePoses: Seq[PossibleTrie] = if (solution.ending.isDefined) {
165             solution.curTrie.map(t => PossibleTrie(t, 0)) ++ koreanPosTrie.map(
166               t => PossibleTrie(t, 1))
167           } else {
168             solution.curTrie.map(t => PossibleTrie(t, 0))
169           }
170 
171           possiblePoses.view.filter { t =>
172             t.curTrie.curPos == Noun || koreanDictionary(t.curTrie.curPos).contains(
173               word.toCharArray)
174           }.map { case t: PossibleTrie =>
175             val candidateToAdd =
176               if (t.curTrie.curPos == Noun && !koreanDictionary(Noun).contains(word.toCharArray)) {
177                 val isWordName: Boolean = isName(word)
178                 val isWordKoreanNameVariation: Boolean = isKoreanNameVariation(word)
179 
180                 val unknown = !isWordName && !isKoreanNumber(word) && !isWordKoreanNameVariation
181                 val pos = Noun
182                 ParsedChunk(Seq(KoreanToken(word, pos, chunk.offset + start, word.length, unknown)),
183                   t.words, profile)
184               } else {
185                 val pos = t.curTrie.curPos
186                 ParsedChunk(Seq(KoreanToken(word, pos, chunk.offset + start, word.length)), t.words,
187                   profile)
188               }
189 
190             val nextTrie = t.curTrie.nextTrie.map {
191               case nt: KoreanPosTrie if nt == selfNode => t.curTrie
192               case nt: KoreanPosTrie => nt
193             }
194 
195             CandidateParse(solution.parse ++ candidateToAdd, nextTrie, t.curTrie.ending)
196           }
197       }
198 
199       val currentSolutions = if (solutions.contains(end)) solutions(end) else List()
200 
201       solutions += end -> (currentSolutions ++ candidates).sortBy {
202         c => (c.parse.score, c.parse.posTieBreaker)
203       }.take(TOP_N_PER_STATE)
204     }
205 
206 
207     val topCandidates: Seq[Seq[KoreanToken]] = if (solutions(chunk.length).isEmpty) {
208       // If the chunk is not parseable, treat it as a unknown noun chunk.
209       Seq(Seq(KoreanToken(chunk.text, Noun, 0, chunk.length, true)))
210     } else {
211       // Return the best parse of the final state
212       solutions(chunk.length).sortBy(c => c.parse.score).map { p => p.parse.posNodes }
213     }
214 
215     (directMatch ++ topCandidates).distinct
216   }

218   private def findDirectMatch(chunk: KoreanToken): Seq[Seq[KoreanToken]] = {
219     // Direct match
220     // This may produce 하 -> PreEomi
221     koreanDictionary.foreach {
222       case (pos, dict) =>
223         if (dict.contains(chunk.text)) {
224           return Seq(Seq(KoreanToken(chunk.text, pos, chunk.offset, chunk.length)))
225         }
226     }
227     Seq()
228   }
