26 object ParsedChunk {
27   val suffixes = Set(Suffix, Eomi, Josa, PreEomi)
28   val preferredBeforeHaVerb = Set(Noun, ProperNoun, VerbPrefix)
29 }
30 
31 /**
32   * A candidate parse for a chunk.
33   *
34   * @param posNodes Sequence of KoreanTokens.
35   * @param words Number of words in this candidate parse.
36   */
37 case class ParsedChunk(posNodes: Seq[KoreanToken], words: Int,
38     profile: TokenizerProfile = TokenizerProfile.defaultProfile) {
39 
40   // Using lazy val to cache the score
41   lazy val score: Float = countTokens * profile.tokenCount +
42       countUnknowns * profile.unknown +
43       words * profile.wordCount +
44       getUnknownCoverage * profile.unknownCoverage +
45       getFreqScore * profile.freq +
46       countPos(Unknown) * profile.unknownPosCount +
47       isExactMatch * profile.exactMatch +
48       isAllNouns * profile.allNoun +
49       isPreferredPattern * profile.preferredPattern +
50       countPos(Determiner) * profile.determinerPosCount +
51       countPos(Exclamation) * profile.exclamationPosCount +
52       isInitialPostPosition * profile.initialPostPosition +
53       isNounHa * profile.haVerb +
54       hasSpaceOutOfGuide * profile.spaceGuidePenalty
55 
56   def countUnknowns: Int = this.posNodes.count { p: KoreanToken => p.unknown }
57 
58   def countTokens: Int = this.posNodes.size
59 
60   def isInitialPostPosition: Int = if (suffixes.contains(this.posNodes.head.pos)) 1 else 0
61 
62   def isExactMatch: Int = if (this.posNodes.size == 1) 0 else 1
63 
64   def hasSpaceOutOfGuide: Int = if (profile.spaceGuide.isEmpty) {
65     0
66   } else {
67     this.posNodes
68         .filter{p: KoreanToken => !suffixes.contains(p.pos)}
69         .count {
70           p: KoreanToken => !profile.spaceGuide.contains(p.offset)
71         }
72   }
73 
74 
75   def isAllNouns: Int = if (this.posNodes.exists(
76     t => t.pos != Noun && t.pos != ProperNoun)) 1
77   else 0
78 
79   def isPreferredPattern: Int = if (
80     posNodes.size == 2 && profile.preferredPatterns.contains(posNodes.map(_.pos))
81   ) 0
82   else 1
83 
84   def isNounHa: Int = if (this.posNodes.size >= 2
85       && preferredBeforeHaVerb.contains(this.posNodes.head.pos)
86       && this.posNodes(1).pos == Verb
87       && (this.posNodes(1).text.startsWith("하") || this.posNodes(1).text.startsWith("해")))
88     0
89   else
90     1
91 
92   def posTieBreaker: Int = this.posNodes.map(_.pos.id).sum
93 
94   def getUnknownCoverage: Int = this.posNodes.foldLeft(0) {
95     case (sum, p: KoreanToken) => if (p.unknown) sum + p.text.length else sum
96   }
97 
98   def getFreqScore: Float = this.posNodes.foldLeft(0f) {
99     case (output: Float, p: KoreanToken) if p.pos == Noun || p.pos == ProperNoun =>
100       output + (1f - koreanEntityFreq.getOrDefault(p.text, 0f))
101     case (output: Float, p: KoreanToken) => output + 1.0f
102   } / this.posNodes.size
103 
104   def ++(that: ParsedChunk): ParsedChunk = {
105     ParsedChunk(this.posNodes ++ that.posNodes, this.words + that.words, profile)
106   }
107 
108   def countPos(pos: KoreanPos): Int = this.posNodes.count { p: KoreanToken => p.pos == pos }
109 }
