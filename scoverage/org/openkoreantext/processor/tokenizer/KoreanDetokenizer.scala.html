24 /**
25   * Detokenizes a list of tokenized words into a readable sentence.
26   */
27 object KoreanDetokenizer {
28   val SuffixPos = Set(KoreanPos.Josa, KoreanPos.Eomi, KoreanPos.PreEomi, KoreanPos.Suffix,
29     KoreanPos.Punctuation)
30   val PrefixPos = Set(KoreanPos.NounPrefix, KoreanPos.VerbPrefix)
31 
32   def detokenize(input: Iterable[String]) = {
33     // Space guide prevents tokenizing a word that was not tokenized in the input.
34     val spaceGuide: Set[Int] = getSpaceGuide(input)
35 
36     // Tokenize a merged text with the space guide.
37     val tokenized = KoreanTokenizer.tokenize(input.mkString(""), TokenizerProfile(spaceGuide = spaceGuide))
38 
39     // Attach suffixes and prefixes.
40     collapseTokens(tokenized).mkString(" ")
41   }
42 
43   private def collapseTokens(tokenized: Seq[KoreanToken]): List[String] = {
44     val (output, isPrefix) = tokenized.foldLeft((List[String](), false)) {
45       case ((output: List[String], isPrefix: Boolean), token: KoreanToken) =>
46         if (output.nonEmpty && (isPrefix || SuffixPos.contains(token.pos))) {
47           val attached = output.lastOption.getOrElse("") + token.text
48           (output.init :+ attached, false)
49         } else if (PrefixPos.contains(token.pos)) {
50           (output :+ token.text, true)
51         } else {
52           (output :+ token.text, false)
53         }
54     }
55     output
56   }
57 
58   private def getSpaceGuide(input: Iterable[String]): Set[Int] = {
59     val (spaceGuide, index) = input.foldLeft((Set[Int](), 0)) {
60       case ((output: Set[Int], i: Int), word: String) =>
61         (output + (i + word.length), i + word.length)
62     }
63     spaceGuide
64   }
65 }
