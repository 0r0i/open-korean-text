26 /**
27  * Create Korean predicate expansion goldenset from adjective and verb resources.
28  * The first argument is the test resource directory.
29  */
30 object CreateConjugationExamples extends Runnable  {
31   case class ConjugationExample(word: String, conjugations: Seq[String])
32 
33   def run {
34     System.err.println("Reading the verbs and adjectives..")
35 
36     def updateConjugateExamples(file: String, isAdj: Boolean, outputFileName: String) {
37       System.err.println("Writing the expansion goldenset in " + outputFileName)
38 
39       val outputPath = "src/test/resources/org/openkoreantext/processor/util/" + outputFileName
40       val out = new FileOutputStream(outputPath)
41 
42       val words = readWordsAsSeq(file)
43       val goldenset = words.map(word =>
44         ConjugationExample(word, conjugatePredicated(Set(word), isAdj).toSeq.sorted)
45       )
46 
47       goldenset.foreach {
48         c => out.write(
49           "%s\t%s\n".format(c.word, c.conjugations.mkString(", ")).getBytes
50         )
51       }
52 
53       out.close()
54     }
55 
56     updateConjugateExamples("adjective/adjective.txt", isAdj = true, "adj_conjugate.txt")
57     updateConjugateExamples("verb/verb.txt", isAdj = false, "verb_conjugate.txt")
58   }
