31     val parsedPairs = readFileByLineFromResources("example_chunks.txt").flatMap {
32       case line if line.length > 0 =>
33         val chunk = line.trim

36           val tokens = p.map{
37             token => KoreanToken(token.text, token.pos, token.offset, token.length)}

39         }.distinct
40         Some(Seq(chunk) ++ parsedString)
41       case line => None
42     }
