27 /**
28  * Create Korean Parsing examples.
29  */
30 object CreateParsingExamples extends Runnable  {
31   case class ParsingExample(text: String, parse: Seq[KoreanToken])
32 
33   def run {
34     System.err.println("Reading the goldenset..")
35 
36     val parsedPairs = readFileByLineFromResources("example_chunks.txt").flatMap {
37       case line if line.length > 0 =>
38         val chunk = line.trim
39         val parsed = tokenize(chunk)
40         Some(ParsingExample(chunk, parsed))
41       case line => None
42     }.toSet
43 
44 
45     val outputFile: String = "src/test/resources/org/openkoreantext/processor/util/current_parsing.txt"
46 
47     System.err.println("Writing the new goldenset to " + outputFile)
48 
49     val out = new FileOutputStream(outputFile)
50     parsedPairs.toSeq.sortBy(_.text).foreach {
51       p =>
52         out.write(p.text.getBytes)
53         out.write("\t".getBytes)
54         out.write(p.parse.mkString("/").getBytes)
55         out.write("\n".getBytes)
56     }
57     out.close()
58 
59     System.err.println("Testing the new goldenset " + outputFile)
60   }
61 }
