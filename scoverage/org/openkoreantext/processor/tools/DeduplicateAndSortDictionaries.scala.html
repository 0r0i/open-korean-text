30   private[this] def readWords(filename: String): Set[String] = {
31     Source.fromFile(filename)(io.Codec("UTF-8"))

33         .map(_.trim)
34         .filter(_.length > 0)
35         .toSet
36   }

58   def run {
59     RESOURCES_TO_CLEANUP.foreach {
60       f: String =>
61         val outputFolder = "src/main/resources/org/openkoreantext/processor/util/"

63         val words = readWords(outputFolder + f).toList.sorted
