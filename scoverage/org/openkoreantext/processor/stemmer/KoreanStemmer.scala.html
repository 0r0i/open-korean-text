7 /**
8  * Stems Adjectives and Verbs: 새로운 스테밍을 추가했었다. -> 새롭다 + 스테밍 + 을 + 추가하다
9  */
10 object KoreanStemmer {
11   private val Endings = Set(Eomi, PreEomi)
12   private val Predicates = Set(Verb, Adjective)
13 
14   private val EndingsForNouns = Set("하다", "되다", "없다")
15 
16   /**
17    * Removes Ending tokens recovering the root form of predicates
18    *
19    * @param tokens A sequence of tokens
20    * @return A sequence of collapsed Korean tokens
21    */
22   def stem(tokens: Seq[KoreanToken]): Seq[KoreanToken] = {
23     if (!tokens.exists(t => t.pos == Verb || t.pos == Adjective)) {
24       return tokens
25     }
26 
27     val stemmed = tokens.foldLeft(List[KoreanToken]()) {
28       case (l: List[KoreanToken], token: KoreanToken) if l.nonEmpty && Endings.contains(token.pos) =>
29         if (Predicates.contains(l.head.pos)) {
30           val prevToken = l.head
31           KoreanToken(
32             prevToken.text + token.text,
33             prevToken.pos, prevToken.offset, prevToken.length + token.length,
34             stem = prevToken.stem,
35             unknown = prevToken.unknown
36           ) :: l.tail
37         } else {
38           l
39         }
40       case (l: List[KoreanToken], token: KoreanToken) if Predicates.contains(token.pos) =>
41         KoreanToken(
42           token.text,
43           token.pos, token.offset, token.length,
44           stem = Some(predicateStems(token.pos)(token.text)),
45           unknown = token.unknown
46         ) :: l
47       case (l: List[KoreanToken], token: KoreanToken) => token :: l
48     }.reverse
49 
50     def validNounHeading(token: KoreanToken): Boolean = {
51       val heading = token.text.take(token.text.length - 2)
52 
53       val validLength = token.text.length > 2
54       val validPos = token.pos == Verb
55       val validEndings = EndingsForNouns.contains(token.text.takeRight(2))
56       val validNouns = koreanDictionary.get(Noun).contains(heading)
57 
58       validLength && validPos && validEndings && validNouns
59     }
60 
61     stemmed.flatMap {
62       case token if validNounHeading(token) =>
63         val heading = token.text.take(token.text.length - 2)
64         val ending = token.text.takeRight(2)
65 
66         Seq(
67           KoreanToken(heading, Noun, token.offset, heading.length),
68           KoreanToken(ending, token.pos, token.offset + heading.length, token.length - heading.length)
69         )
70       case token => Seq(token)
71     }
72   }
73 }
