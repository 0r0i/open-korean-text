31 /**
32  * Normalize Korean colloquial text
33  */
34 object KoreanNormalizer {
35   private[this] val EXTENTED_KOREAN_REGEX = """([ㄱ-ㅣ가-힣]+)""".r
36   private[this] val KOREAN_TO_NORMALIZE_REGEX = """([가-힣]+)(ㅋ+|ㅎ+|[ㅠㅜ]+)""".r
37   protected[processor] val REPEATING_CHAR_REGEX: Regex = """(.)\1{3,}|[ㅠㅜ]{3,}""".r
38   private[this] val REPEATING_2CHAR_REGEX = """(..)\1{2,}""".r
39 
40   private[this] val WHITESPACE_REGEX = """\s+""".r
41 
42   private[this] case class Segment(text: String, matchData: Option[Match])
43 
44   private[this] val CODA_N_EXCPETION = "은는운인텐근른픈닌든던".toSet
45 
46   /**
47    * Normalize Korean CharSequence text
48    * ex) 하댘ㅋㅋㅋ -> 하대, 머구뮤ㅠㅠㅠ -> 머굼
49    * 하즤 -> 하지
50    *
51    * @param input input CharSequence
52    * @return normalized CharSequence
53    */
54   def normalize(input: CharSequence): CharSequence = {
55     EXTENTED_KOREAN_REGEX.replaceAllIn(input, m => normalizeKoreanChunk(m.group(0)).toString)
56   }
57 
58   private[this] def normalizeKoreanChunk(input: CharSequence): CharSequence = {
59     // Normalize endings: 안됔ㅋㅋㅋ -> 안돼ㅋㅋ
60     val endingNormalized = KOREAN_TO_NORMALIZE_REGEX.replaceAllIn(
61       input, m => processNormalizationCandidate(m).toString
62     )
63 
64     // Normalize repeating chars: ㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋ -> ㅋㅋㅋ
65     val exclamationNormalized = REPEATING_CHAR_REGEX.replaceAllIn(
66       endingNormalized, m => {
67         Matcher.quoteReplacement(m.group(0).take(3).toString)
68       }
69     )
70     // Normalize repeating chars: 훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍 -> 훌쩍훌쩍
71     val repeatingNormalized = REPEATING_2CHAR_REGEX.replaceAllIn(
72       exclamationNormalized, m => {
73         Matcher.quoteReplacement(m.group(0).take(4).toString)
74       }
75     )
76 
77     // Coda normalization (명사 + ㄴ 첨가 정규화): 소린가 -> 소리인가
78     val codaNNormalized = normalizeCodaN(repeatingNormalized)
79     // Typo correction: 하겟다 -> 하겠다
80     val typoCorrected = correctTypo(codaNNormalized)
81     // Spaces, tabs, new lines are replaced with a single space.
82     WHITESPACE_REGEX.replaceAllIn(typoCorrected, " ")
83   }

85   protected[processor] def correctTypo(chunk: CharSequence): CharSequence =
86     typoDictionaryByLength.foldLeft(chunk) {
87       case (output: String, (wordLen: Int, typoMap: Map[String, String])) =>
88         output.sliding(wordLen).foldLeft(output) {
89           case (sliceOutput: String, slice: String) if typoMap.contains(slice) =>
90             sliceOutput.replaceAll(slice, typoMap(slice))
91           case (sliceOutput: String, slice: String) =>
92             sliceOutput
93         }
94     }
95 
96   protected[processor] def normalizeCodaN(chunk: CharSequence): CharSequence = {
97     if (chunk.length < 2) return chunk
98 
99     val lastTwo = chunk.subSequence(chunk.length() - 2, chunk.length())
100     val last = chunk.charAt(chunk.length() - 1)
101 
102     val lastTwoHead = lastTwo.charAt(0)
103 
104     // Exception cases
105     if (koreanDictionary.get(Noun).contains(chunk) ||
106         koreanDictionary.get(Conjunction).contains(chunk) ||
107         koreanDictionary.get(Adverb).contains(chunk) ||
108         koreanDictionary.get(Noun).contains(lastTwo) ||
109         lastTwoHead < '가' || lastTwoHead > '힣' ||
110         CODA_N_EXCPETION.contains(lastTwoHead)
111     ) {
112       return chunk
113     }
114 
115     val hc = decomposeHangul(lastTwoHead)
116 
117     val newHead = new StringBuilder()
118         .append(chunk.subSequence(0, chunk.length() - 2))
119         .append(composeHangul(hc.onset, hc.vowel))
120 
121     if (hc.coda == 'ㄴ' &&
122         (last == '데' || last == '가' || last == '지') &&
123         koreanDictionary.get(Noun).contains(newHead)
124     ) {
125       newHead + "인" + last
126     } else {
127       chunk
128     }
129   }
130 
131   private[this] def processNormalizationCandidate(m: Match): CharSequence = {
132     val chunk = m.group(1)
133     val toNormalize = m.group(2)
134 
135     val normalizedChunk = if (koreanDictionary.get(Noun).contains(chunk) ||
136         koreanDictionary.get(Eomi).contains(chunk.takeRight(1)) ||
137         koreanDictionary.get(Eomi).contains(chunk.takeRight(2))) {
138       chunk
139     } else {
140       normalizeEmotionAttachedChunk(chunk, toNormalize)
141     }
142     normalizedChunk + toNormalize
143   }
144 
145   private[this] def normalizeEmotionAttachedChunk(s: CharSequence, toNormalize: CharSequence): CharSequence = {
146     val init = s.subSequence(0, s.length() - 1)
147     val secondToLastDecomposed = init match {
148       case si: CharSequence if si.length > 0 =>
149         val hc = decomposeHangul(si.charAt(si.length() - 1))
150         if (hc.coda == ' ') Some(hc) else None
151       case _ => None
152     }
153 
154     decomposeHangul(s.charAt(s.length() - 1)) match {
155       case hc: HangulChar if hc.coda == 'ㅋ' || hc.coda == 'ㅎ' =>
156         new StringBuilder()
157             .append(init)
158             .append(composeHangul(hc.onset, hc.vowel))
159       case HangulChar(o: Char, v: Char, ' ') if secondToLastDecomposed.isDefined &&
160           (v == toNormalize.charAt(0)) &&
161           Hangul.CODA_MAP.contains(o) =>
162         val hc = secondToLastDecomposed.get
163         new StringBuilder()
164             .append(init.subSequence(0, init.length() - 1))
165             .append(composeHangul(hc.onset, hc.vowel, o))
166       case _ => s
167     }
168   }
