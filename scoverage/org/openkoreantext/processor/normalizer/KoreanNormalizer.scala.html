33 object KoreanNormalizer {
34   private[this] val EXTENTED_KOREAN_REGEX = """([ㄱ-ㅣ가-힣]+)""".r
35   private[this] val KOREAN_TO_NORMALIZE_REGEX = """([가-힣]+)(ㅋ+|ㅎ+|[ㅠㅜ]+)""".r
36   protected[processor] val REPEATING_CHAR_REGEX = """(.)\1{2,}|[ㅠㅜ]{2,}""".r
37   private[this] val REPEATING_2CHAR_REGEX = """(..)\1{2,}""".r
38 
39   private[this] val WHITESPACE_REGEX = """\s+""".r
40 
41   private[this] case class Segment(text: String, matchData: Option[Match])
42 
43   private[this] val CODA_N_EXCPETION = "은는운인텐근른픈닌든던".toSet

45   /**
46    * Normalize Korean CharSequence text
47    * ex) 하댘ㅋㅋㅋ -> 하대, 머구뮤ㅠㅠㅠ -> 머굼
48    * 하즤 -> 하지
49    *
50    * @param input input CharSequence
51    * @return normalized CharSequence
52    */
53   def normalize(input: CharSequence): CharSequence = {
54     EXTENTED_KOREAN_REGEX.replaceAllIn(input, m => normalizeKoreanChunk(m.group(0)).toString)
55   }
56 
57   private[this] def normalizeKoreanChunk(input: CharSequence): CharSequence = {
58     // Normalize endings: 안됔ㅋㅋㅋ -> 안돼ㅋㅋ
59     val endingNormalized = KOREAN_TO_NORMALIZE_REGEX.replaceAllIn(
60       input, m => processNormalizationCandidate(m).toString
61     )
62 
63     // Normalize repeating chars: ㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋ -> ㅋㅋ
64     val exclamationNormalized = REPEATING_CHAR_REGEX.replaceAllIn(
65       endingNormalized, m => {
66         Matcher.quoteReplacement(m.group(0).take(2).toString)
67       }
68     )
69     // Normalize repeating chars: 훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍 -> 훌쩍훌쩍
70     val repeatingNormalized = REPEATING_2CHAR_REGEX.replaceAllIn(
71       exclamationNormalized, m => {
72         Matcher.quoteReplacement(m.group(0).take(4).toString)
73       }
74     )
75 
76     // Coda normalization (명사 + ㄴ 첨가 정규화): 소린가 -> 소리인가
77     val codaNNormalized = normalizeCodaN(repeatingNormalized)
78     // Typo correction: 하겟다 -> 하겠다
79     val typoCorrected = correctTypo(codaNNormalized)
80     // Spaces, tabs, new lines are replaced with a single space.
81     WHITESPACE_REGEX.replaceAllIn(typoCorrected, " ")
82   }

84   protected[processor] def correctTypo(chunk: CharSequence): CharSequence = {
85     typoDictionaryByLength.foldLeft(chunk) {
86       case (output: String, (wordLen: Int, typoMap: Map[String, String])) =>
87         output.sliding(wordLen).foldLeft(output) {
88           case (sliceOutput: String, slice: String) if typoMap.contains(slice) =>
89             sliceOutput.replaceAll(slice, typoMap(slice))
90           case (sliceOutput: String, slice: String) =>
91             sliceOutput
92         }
93     }
94   }
95 
96   protected[processor] def normalizeCodaN(chunk: CharSequence): CharSequence = {
97     if (chunk.length < 2) return chunk
98 
99     val lastTwo = chunk.subSequence(chunk.length() - 2, chunk.length())
100     val last = chunk.charAt(chunk.length() - 1)
101 
102     val lastTwoHead = lastTwo.charAt(0)
103 
104     // Exception cases
105     if (koreanDictionary(Noun).contains(chunk) ||
106         koreanDictionary(Conjunction).contains(chunk) ||
107         koreanDictionary(Adverb).contains(chunk) ||
108         koreanDictionary(Noun).contains(lastTwo) ||
109         lastTwoHead < '가' || lastTwoHead > '힣' ||
110         CODA_N_EXCPETION.contains(lastTwoHead)
111     ) {
112       return chunk
113     }
114 
115     val hc = decomposeHangul(lastTwoHead)
116 
117     val newHead = new StringBuilder()
118         .append(chunk.subSequence(0, chunk.length() - 2))
119         .append(composeHangul(hc.onset, hc.vowel))
120 
121     if (hc.coda == 'ㄴ' &&
122         (last == '데' || last == '가' || last == '지') &&
123         koreanDictionary(Noun).contains(newHead)
124     ) {
125       val mid = if (hc.vowel == 'ㅡ') "은" else "인"
126       newHead + mid + last
127     } else {
128       chunk
129     }
130   }
131 
132   private[this] def processNormalizationCandidate(m: Match): CharSequence = {
133     val chunk = m.group(1)
134     val toNormalize = m.group(2)
135 
136     val normalizedChunk = if (koreanDictionary(Noun).contains(chunk) ||
137         koreanDictionary(Eomi).contains(chunk.takeRight(1)) ||
138         koreanDictionary(Eomi).contains(chunk.takeRight(2))) {
139       chunk
140     } else {
141       normalizeEmotionAttachedChunk(chunk, toNormalize)
142     }
143     normalizedChunk + toNormalize
144   }
145 
146   private[this] def normalizeEmotionAttachedChunk(s: CharSequence, toNormalize: CharSequence): CharSequence = {
147     val init = s.subSequence(0, s.length() - 1)
148     val secondToLastDecomposed = init match {
149       case si: CharSequence if si.length > 0 =>
150         val hc = decomposeHangul(si.charAt(si.length() - 1))
151         if (hc.coda == ' ') Some(hc) else None
152       case _ => None
153     }
154 
155     decomposeHangul(s.charAt(s.length() - 1)) match {
156       case hc: HangulChar if hc.coda == 'ㅋ' || hc.coda == 'ㅎ' =>
157         new StringBuilder()
158             .append(init)
159             .append(composeHangul(hc.onset, hc.vowel))
160       case HangulChar(o: Char, v: Char, ' ') if secondToLastDecomposed.isDefined &&
161           (v == toNormalize.charAt(0)) &&
162           Hangul.CODA_MAP.contains(o) =>
163         val hc = secondToLastDecomposed.get
164         new StringBuilder()
165             .append(init.subSequence(0, init.length() - 1))
166             .append(composeHangul(hc.onset, hc.vowel, o))
167       case _ => s
168     }
169   }
